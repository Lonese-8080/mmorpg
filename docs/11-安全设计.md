## 1. 安全设计原则

### 1.1 核心原则

```
┌─────────────────────────────────────────────────────────────────┐
│                    🎯 安全设计的黄金法则                           │
│                                                                 │
│  ┌─────────────────────────────────────────────────────────┐  │
│  │  1. 永远不要相信客户端                                     │  │
│  │     - 客户端发送的所有数据都必须验证                        │  │
│  │     - 客户端可以被修改、调试、劫持                          │  │
│  │     - 任何"只靠客户端验证"都是不安全的                      │  │
│  └─────────────────────────────────────────────────────────┘  │
│                                                                 │
│  ┌─────────────────────────────────────────────────────────┐  │
│  │  2. 防御纵深 (Defense in Depth)                          │  │
│  │     - 不要依赖单一的安全措施                               │  │
│  │     - 每层验证都很重要                                     │  │
│  │     - 层层设防，攻破一层还有下一层                         │  │
│  └─────────────────────────────────────────────────────────┘  │
│                                                                 │
│  ┌─────────────────────────────────────────────────────────┐  │

│  └─────────────────────────────────────────────────────────┘  │
│                                                                 │
│  ┌─────────────────────────────────────────────────────────┐  │
│  │  4. 记录一切 (Log Everything)                            │  │
│  │     - 记录所有可疑行为                                    │  │
│  │     - 记录所有关键操作                                    │  │
│  │     - 可追溯才能可防御                                    │  │
│  └─────────────────────────────────────────────────────────┘  │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

### 2.2 移动反作弊（防瞬移）

```csharp
/// <summary>
/// 移动验证器 - 防止瞬移外挂
/// 
/// 工作原理：
/// 1. 记录玩家每条移动消息的时间戳和位置
/// 2. 验证移动距离是否合理（根据速度上限和时间差）
/// 3. 验证路径是否可通行（不能穿墙）
/// 4. 发现异常时记录并处理
/// </summary>
public class MovementValidator
{
    // ============================================================
    // 配置
    // ============================================================
    
    /// <summary>
    /// 最大行走速度（米/秒）
    /// 正常玩家跑步速度应该在 5-10 米/秒
    /// </summary>
    private const float MaxWalkSpeed = 10.0f;
    
    /// <summary>
    /// 最大冲刺速度（米/秒）
    /// 某些技能可能临时提升速度
    /// </summary>
    private const float MaxSprintSpeed = 15.0f;
    
    /// <summary>
    /// 允许的最大跳跃距离（米）
    /// </summary>
    private const float MaxJumpDistance = 5.0f;
    
    /// <summary>
    /// 速度容差（百分比）
    /// 允许超出上限 10%，防止误杀正常玩家
    /// </summary>
    private const float SpeedTolerance = 1.1f;
    
    // ============================================================
    // 字段
    // ============================================================
    
    /// <summary>
    /// 玩家上次验证的位置
    /// </summary>
    private readonly ConcurrentDictionary<long, PlayerMovementState> _playerStates;
    
    /// <summary>
    /// 场景管理器（用于碰撞检测）
    /// </summary>
    private readonly ISceneManager _sceneManager;
    
    /// <summary>
    /// 告警管理器
    /// </summary>
    private readonly IAntiCheatAlert _alertManager;
    
    // ============================================================
    // 验证逻辑
    // ============================================================
    
    /// <summary>
    /// 验证玩家移动请求
    /// 
    /// 这是服务器端的核心验证点
    /// 所有的移动都必须经过这个验证
    /// </summary>
    /// <param name="session">玩家会话</param>
    /// <param name="request">移动请求</param>
    /// <returns>验证结果</returns>
    public MovementValidationResult Validate(
        Session session, 
        C2S_Move request)
    {
        var playerId = session.PlayerId;
        var playerState = GetOrCreateState(playerId);
        
        // ========================================================
        // 步骤1：时间验证
        // ========================================================
        
        var currentTime = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds();
        var timeDelta = currentTime - playerState.LastMoveTime;
        
        // 防止时间回退攻击（客户端时间可能被修改）
        if (timeDelta < 0)
        {
            Logger.Warning($"玩家 {playerId} 时间回退检测: delta={timeDelta}ms");
            return MovementValidationResult.Invalid("客户端时间异常");
        }
        
        // 防止加速器（时间间隔太小）
        // 正常情况下客户端至少每 100ms 发一次移动消息
        if (timeDelta < 50)
        {
            Logger.Warning($"玩家 {playerId} 消息频率异常: delta={timeDelta}ms");
            _alertManager.Alert(playerId, AlertType.FrequencyAnomaly, 
                $"移动频率异常: {timeDelta}ms");
        }
        
        // ========================================================
        // 步骤2：速度验证
        // ========================================================
        
        var currentPos = playerState.CurrentPosition;
        var targetPos = new Vector3(request.Target.X, request.Target.Y, request.Target.Z);
        
        // 计算移动距离
        var distance = Vector3.Distance(currentPos, targetPos);
        
        // 计算允许的最大距离
        var maxSpeed = GetPlayerMaxSpeed(playerId); // 考虑BUFF、状态等
        var maxDistance = maxSpeed * (timeDelta / 1000.0f) * SpeedTolerance;
        
        if (distance > maxDistance)
        {
            Logger.Warning(
                $"玩家 {playerId} 速度异常: " +
                $"距离={distance:F2}m, 允许={maxDistance:F2}m, " +
                $"时间差={timeDelta}ms, 速度={maxSpeed}m/s");
            
            _alertManager.Alert(playerId, AlertType.SpeedHack, 
                $"瞬移检测: 距离{distance:F2}m, 时间{timeDelta}ms");
            
            return MovementValidationResult.Invalid("移动速度异常");
        }
        
        // ========================================================
        // 步骤3：碰撞检测（穿墙验证）
        // ========================================================
        
        var scene = _sceneManager.GetScene(playerState.SceneId);
        if (!scene.IsPathValid(currentPos, targetPos))
        {
            Logger.Warning($"玩家 {playerId} 穿墙检测: " +
                $"从 {currentPos} 到 {targetPos}");
            
            _alertManager.Alert(playerId, AlertType.WallHack,
                $"穿墙检测: 路径不合法");
            
            return MovementValidationResult.Invalid("路径不合法");
        }
        
        // ========================================================
        // 步骤4：区域验证（不能进入禁止区域）
        // ========================================================
        
        if (scene.IsForbiddenZone(targetPos))
        {
            Logger.Warning($"玩家 {playerId} 进入禁止区域: {targetPos}");
            
            _alertManager.Alert(playerId, AlertType.ForbiddenZone,
                $"进入禁止区域: {targetPos}");
            
            return MovementValidationResult.Invalid("禁止进入该区域");
        }
        
        // ========================================================
        // 步骤5：更新状态
        // ========================================================
        
        playerState.CurrentPosition = targetPos;
        playerState.LastMoveTime = currentTime;
        playerState.MoveCount++;
        
        return MovementValidationResult.Valid;
    }
    
    /// <summary>
    /// 获取玩家当前状态（带锁保护）
    /// </summary>
    private PlayerMovementState GetOrCreateState(long playerId)
    {
        return _playerStates.GetOrAdd(playerId, _ => new PlayerMovementState
        {
            LastMoveTime = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds()
        });
    }
    
    /// <summary>
    /// 获取玩家最大速度（考虑各种因素）
    /// </summary>
    private float GetPlayerMaxSpeed(long playerId)
    {
        // TODO: 从玩家属性、状态、BUFF中计算
        // 目前返回基础速度
        return MaxWalkSpeed;
    }
}

/// <summary>
/// 玩家移动状态
/// </summary>
public class PlayerMovementState
{
    /// <summary>
    /// 当前坐标
    /// </summary>
    public Vector3 CurrentPosition;
    
    /// <summary>
    /// 最后移动时间（毫秒时间戳）
    /// </summary>
    public long LastMoveTime;
    
    /// <summary>
    /// 当前场景ID
    /// </summary>
    public int SceneId;
    
    /// <summary>
    /// 移动计数（用于统计）
    /// </summary>
    public long MoveCount;
}

/// <summary>
/// 移动验证结果
/// </summary>
public class MovementValidationResult
{
    public bool IsValid { get; private set; }
    public string Reason { get; private set; }
    
    public static MovementValidationResult Valid => 
        new() { IsValid = true };
    
    public static MovementValidationResult Invalid(string reason) => 
        new() { IsValid = false, Reason = reason };
}
```

### 2.3 战斗反作弊（防秒杀、防连击）

```csharp
/// <summary>
/// 战斗验证器 - 防止秒杀外挂、连击外挂
/// 
/// 工作原理：
/// 1. 验证攻击间隔（防止超频攻击）
/// 2. 验证攻击范围（防止超远攻击）
/// 3. 验证技能冷却（防止无限技能）
/// 4. 验证伤害值（防止秒杀）
/// 5. 验证目标有效性（防止空气攻击）
/// </summary>
public class BattleValidator
{
    // ============================================================
    // 配置
    // ============================================================
    
    /// <summary>
    /// 最小攻击间隔（毫秒）
    /// 正常攻击至少间隔 500ms
    /// </summary>
    private const int MinAttackInterval = 500;
    
    /// <summary>
    /// 最大攻击距离（米）
    /// 超出这个距离的攻击无效
    /// </summary>
    private const float MaxAttackRange = 10.0f;
    
    /// <summary>
    /// 伤害容差（百分比）
    /// 允许实际伤害超出计算伤害的容差范围
    /// </summary>
    private const float DamageTolerance = 1.2f;
    
    // ============================================================
    // 验证逻辑
    // ============================================================
    
    /// <summary>
    /// 验证普通攻击
    /// </summary>
    public BattleValidationResult ValidateNormalAttack(
        Session session,
        C2S_NormalAttack request)
    {
        var playerId = session.PlayerId;
        var playerState = GetPlayerState(playerId);
        
        // ========================================================
        // 步骤1：攻击间隔验证
        // ========================================================
        
        var currentTime = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds();
        var timeDelta = currentTime - playerState.LastAttackTime;
        
        if (timeDelta < MinAttackInterval)
        {
            Logger.Warning($"玩家 {playerId} 攻击频率异常: " +
                $"间隔={timeDelta}ms, 最小={MinAttackInterval}ms");
            
            Alert(playerId, AlertType.AttackSpeedHack,
                $"攻击超频: 间隔{timeDelta}ms");
            
            return BattleValidationResult.Invalid("攻击间隔太短");
        }
        
        // ========================================================
        // 步骤2：目标验证
        // ========================================================
        
        var targetEntity = World.Instance.GetEntity(request.TargetId);
        if (targetEntity == null || !targetEntity.IsActive)
        {
            return BattleValidationResult.Invalid("目标不存在");
        }
        
        // ========================================================
        // 步骤3：距离验证
        // ========================================================
        
        var attackerPos = playerState.CurrentPosition;
        var targetPos = targetEntity.GetPosition();
        var distance = Vector3.Distance(attackerPos, targetPos);
        
        if (distance > MaxAttackRange)
        {
            Logger.Warning($"玩家 {playerId} 攻击距离异常: " +
                $"距离={distance:F2}m, 最大={MaxAttackRange}m");
            
            Alert(playerId, AlertType.AttackRangeHack,
                $"超远攻击: 距离{distance:F2}m");
            
            return BattleValidationResult.Invalid("目标距离太远");
        }
        
        // ========================================================
        // 步骤4：伤害计算与验证
        // ========================================================
        
        var calculatedDamage = CalculateDamage(playerId, request.TargetId);
        var reportedDamage = request.ReportedDamage; // 客户端报告的伤害
        
        // 允许一定容差
        if (reportedDamage > calculatedDamage * DamageTolerance)
        {
            Logger.Warning($"玩家 {playerId} 伤害异常: " +
                $"报告={reportedDamage}, 计算={calculatedDamage}");
            
            Alert(playerId, AlertType.DamageHack,
                $"伤害异常: 报告{reportedDamage}, 计算{calculatedDamage}");
            
            return BattleValidationResult.Invalid("伤害值异常");
        }
        
        // ========================================================
        // 步骤5：更新状态
        // ========================================================
        
        playerState.LastAttackTime = currentTime;
        playerState.AttackCount++;
        
        return BattleValidationResult.Valid;
    }
    
    /// <summary>
    /// 验证技能释放
    /// </summary>
    public BattleValidationResult ValidateSkill(
        Session session,
        C2S_UseSkill request)
    {
        var playerId = session.PlayerId;
        var playerState = GetPlayerState(playerId);
        
        // ========================================================
        // 步骤1：技能是否存在
        // ========================================================
        
        var skillConfig = SkillManager.Instance.GetConfig(request.SkillId);
        if (skillConfig == null)
        {
            Logger.Warning($"玩家 {playerId} 使用不存在的技能: {request.SkillId}");
            return BattleValidationResult.Invalid("技能不存在");
        }
        
        // ========================================================
        // 步骤2：技能冷却验证
        // ========================================================
        
        var cooldownEndTime = playerState.GetSkillCooldown(request.SkillId);
        var currentTime = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds();
        
        if (currentTime < cooldownEndTime)
        {
            var remainingCooldown = cooldownEndTime - currentTime;
            Logger.Warning($"玩家 {playerId} 技能冷却中: " +
                $"技能={request.SkillId}, 剩余={remainingCooldown}ms");
            
            Alert(playerId, AlertType.CooldownHack,
                $"技能超频: 技能{request.SkillId}, 冷却剩余{remainingCooldown}ms");
            
            return BattleValidationResult.Invalid("技能冷却中");
        }
        
        // ========================================================
        // 步骤3：魔法值验证
        // ========================================================
        
        var playerMp = playerState.CurrentMp;
        if (playerMp < skillConfig.MpCost)
        {
            Logger.Warning($"玩家 {playerId} 魔法值不足: " +
                $"当前={playerMp}, 需要={skillConfig.MpCost}");
            
            return BattleValidationResult.Invalid("魔法值不足");
        }
        
        // ========================================================
        // 步骤4：距离验证（如果是指向性技能）
        // ========================================================
        
        if (request.TargetId > 0) // 有目标
        {
            var targetEntity = World.Instance.GetEntity(request.TargetId);
            if (targetEntity != null)
            {
                var distance = Vector3.Distance(
                    playerState.CurrentPosition, 
                    targetEntity.GetPosition());
                
                if (distance > skillConfig.MaxRange)
                {
                    Logger.Warning($"玩家 {playerId} 技能距离异常: " +
                        $"技能={request.SkillId}, 距离={distance:F2}m");
                    
                    return BattleValidationResult.Invalid("目标距离太远");
                }
            }
        }
        
        // ========================================================
        // 步骤5：更新冷却
        // ========================================================
        
        playerState.SetSkillCooldown(request.SkillId, 
            currentTime + skillConfig.Cooldown);
        playerState.CurrentMp -= skillConfig.MpCost;
        
        return BattleValidationResult.Valid;
    }
    
    /// <summary>
    /// 计算伤害值
    /// </summary>
    private int CalculateDamage(long attackerId, long targetId)
    {
        // 从 ECS 获取属性
        var attacker = World.Instance.GetEntity(attackerId);
        var target = World.Instance.GetEntity(targetId);
        
        if (attacker == null || target == null)
            return 0;
        
        var attack = attacker.GetComponent<AttackComponent>();
        var defense = target.GetComponent<DefenseComponent>();
        
        // 简化伤害计算公式
        var damage = attack.Attack - defense.Defense;
        damage = Math.Max(1, damage); // 至少造成1点伤害
        
        return damage;
    }
}

/// <summary>
/// 战斗验证结果
/// </summary>
public class BattleValidationResult
{
    public bool IsValid { get; private set; }
    public string Reason { get; private set; }
    
    public static BattleValidationResult Valid => 
        new() { IsValid = true };
    
    public static BattleValidationResult Invalid(string reason) => 
        new() { IsValid = false, Reason = reason };
}
```

### 2.4 消息限流器（防DoS）

> **源码**：`MMORPG.Framework.Security.RateLimiter`
>
> **算法**：令牌桶（Token Bucket）。以每秒为单位补充令牌，令牌数不足时拒绝消息。

```csharp
/// <summary>
/// 消息限流器 - 令牌桶算法
///
/// 工作原理：
/// 1. 初始化时指定每秒最大消息数（maxMessagesPerSecond）
/// 2. 每收到一条消息尝试获取一个令牌
/// 3. 令牌数量不足时拒绝消息
/// 4. 令牌每秒自动补充到最大容量
/// </summary>
public class RateLimiter
{
    /// <summary>构造限流器</summary>
    /// <param name="maxMessagesPerSecond">每秒最大消息数</param>
    public RateLimiter(long maxMessagesPerSecond);

    /// <summary>尝试获取一个令牌（线程安全）</summary>
    /// <returns>true = 允许处理，false = 超过限流</returns>
    public bool TryAcquire();

    /// <summary>尝试获取 count 个令牌（线程安全）</summary>
    public bool TryAcquire(int count);

    /// <summary>当前可用令牌数（线程安全）</summary>
    public long CurrentTokens { get; }

    /// <summary>总共拒绝的消息数（线程安全）</summary>
    public long DroppedCount { get; }

    /// <summary>最后一次拒绝消息的时间（UTC）</summary>
    public DateTime LastDroppedTime { get; }

    /// <summary>每秒最大消息数</summary>
    public long MaxMessagesPerSecond { get; }

    /// <summary>重置限流器状态（主要用于测试）</summary>
    public void Reset();
}
```

**使用示例**：

```csharp
// 会话级：单玩家 100 TPS
var sessionLimiter = new RateLimiter(100);
if (sessionLimiter.TryAcquire())
{
    // 处理消息
}
else
{
    // 丢弃消息（可记录日志）
}

// 全局级：整服 10000 TPS

```

**与 MessageRouter 的集成**：
- `MessageRouter.SetGlobalRateLimit(long)` 内部使用 `RateLimiter`
- `Session.SetSessionRateLimit(long)` 内部使用 `RateLimiter`
- `FrameworkBootstrap.Configure` 会根据 `FrameworkOptions.EnableGlobalRateLimiting` 自动初始化

### 3.2 密钥交换流程

```csharp
/// <summary>
/// 密钥交换管理器
/// 
/// 使用 ECDH (Elliptic Curve Diffie-Hellman) 进行密钥交换
/// 流程：
/// 1. 服务器生成密钥对 (公钥 + 私钥)
/// 2. 客户端生成密钥对
/// 3. 双方交换公钥
/// 4. 各自使用对方的公钥和自己的私钥计算出共享密钥
/// 5. 使用共享密钥进行 AES 加密通信
/// </summary>
public class KeyExchangeManager
{
    // ============================================================
    // 字段
    // ============================================================
    
    /// <summary>
    /// 服务器私钥
    /// </summary>
    private readonly byte[] _serverPrivateKey;
    
    /// <summary>
    /// 服务器公钥
    /// </summary>
    private readonly byte[] _serverPublicKey;
    
    /// <summary>
    /// 玩家会话密钥
    /// Key: PlayerId
    /// Value: SessionKey
    /// </summary>
    private readonly ConcurrentDictionary<long, SessionKey> _sessionKeys = new();
    
    // ============================================================
    // 初始化
    // ============================================================
    
    /// <summary>
    /// 初始化密钥管理器
    /// </summary>
    public KeyExchangeManager()
    {
        // 生成服务器密钥对
        // 使用 ECDH P-256 曲线
        var keyParams = ECCurve.CreateByName("NIST P-256");
        var key = ECDiffieHellman.Create(keyParams);
        
        _serverPrivateKey = key.ExportECPrivateKey();
        _serverPublicKey = key.ExportSubjectPublicKeyInfo();
    }
    
    // ============================================================
    // 密钥交换
    // ============================================================
    
    /// <summary>
    /// 处理客户端密钥请求
    /// 
    /// 这是登录流程的一部分
    /// 在验证账号密码成功后执行
    /// </summary>
    /// <param name="session">玩家会话</param>
    /// <param name="clientPublicKey">客户端公钥</param>
    /// <returns>服务器的公钥（供客户端完成密钥交换）</returns>
    public byte[] HandleKeyExchange(Session session, byte[] clientPublicKey)
    {
        try
        {
            // ========================================================
            // 步骤1：导入客户端公钥
            // ========================================================
            
            var clientKey = ECDiffieHellman.Create();
            clientKey.ImportSubjectPublicKeyInfo(clientPublicKey, out _);
            
            // ========================================================
            // 步骤2：计算共享密钥
            // ========================================================
            
            var serverKey = ECDiffieHellman.Create();
            serverKey.ImportECPrivateKey(_serverPrivateKey, out _);
            
            var sharedSecret = serverKey.DeriveKeyMaterial(clientKey.PublicKey);
            
            // ========================================================
            // 步骤3：派生会话密钥
            // ========================================================
            
            // 使用 HKDF 派生多个密钥
            var sessionKey = DeriveSessionKeys(sharedSecret, session.ConnectionId);
            
            // ========================================================
            // 步骤4：保存会话密钥
            // ========================================================
            
            _sessionKeys[session.PlayerId] = sessionKey;
            
            // ========================================================
            // 步骤5：清理临时数据
            // ========================================================
            
            Array.Clear(sharedSecret, 0, sharedSecret.Length);
            
            Logger.Info($"玩家 {session.PlayerId} 密钥交换完成，会话密钥已建立");
            
            return _serverPublicKey;
        }
        catch (Exception ex)
        {
            Logger.Error(ex, $"密钥交换失败: 玩家 {session.PlayerId}");
            throw new SecurityException("密钥交换失败", ex);
        }
    }
    
    /// <summary>
    /// 使用 HKDF 派生会话密钥
    /// 
    /// 从共享密钥派生出多个用途不同的密钥：
    /// - 加密密钥 (Encryption Key)
    /// - MAC 密钥 (Message Authentication Key)
    /// - 初始向量 (IV)
    /// </summary>
    private SessionKey DeriveSessionKeys(byte[] sharedSecret, long connectionId)
    {
        // 使用固定盐值（可以加入 connectionId 增加随机性）
        var salt = BitConverter.GetBytes(connectionId);
        
        // 派生加密密钥 (256位 = 32字节)
        var encryptKey = HKDF.DeriveKey(
            HashAlgorithmName.SHA256,
            sharedSecret,
            32,
            salt,
            info: "MMORPG-Encryption-Key"
        );
        
        // 派生 MAC 密钥 (256位 = 32字节)
        var macKey = HKDF.DeriveKey(
            HashAlgorithmName.SHA256,
            sharedSecret,
            32,
            salt,
            info: "MMORPG-MAC-Key"
        );
        
        return new SessionKey
        {
            EncryptKey = encryptKey,
            MacKey = macKey,
            ConnectionId = connectionId,
            CreatedAt = DateTimeOffset.UtcNow
        };
    }
    
    /// <summary>
    /// 获取玩家的会话密钥
    /// </summary>
    public SessionKey? GetSessionKey(long playerId)
    {
        _sessionKeys.TryGetValue(playerId, out var key);
        return key;
    }
    
    /// <summary>
    /// 清除玩家的会话密钥（登出时）
    /// </summary>
    public void ClearSessionKey(long playerId)
    {
        if (_sessionKeys.TryRemove(playerId, out var key))
        {
            // 安全清理
            Array.Clear(key.EncryptKey, 0, key.EncryptKey.Length);
            Array.Clear(key.MacKey, 0, key.MacKey.Length);
        }
    }
}

/// <summary>
/// 会话密钥
/// </summary>
public class SessionKey
{
    /// <summary>
    /// 加密密钥
    /// </summary>
    public byte[] EncryptKey { get; set; }
    
    /// <summary>
    /// MAC 密钥
    /// </summary>
    public byte[] MacKey { get; set; }
    
    /// <summary>
    /// 连接ID
    /// </summary>
    public long ConnectionId { get; set; }
    
    /// <summary>
    /// 创建时间
    /// </summary>
    public DateTimeOffset CreatedAt { get; set; }
}
```

### 3.3 消息加密和解密

```csharp
/// <summary>
/// 消息加密器
/// 
/// 使用 AES-256-GCM 进行消息加密
/// GCM 模式提供了加密和完整性验证
/// 
/// 消息格式：
/// ┌──────────┬────────────┬──────────┬────────────┐
/// │  Nonce   │  加密数据   │  AuthTag │    MAC     │
/// │  (12B)   │  (N B)     │  (16B)  │   (32B)   │
/// └──────────┴────────────┴──────────┴────────────┘
/// </summary>
public class MessageEncryptor
{
    // ============================================================
    // 常量
    // ============================================================
    
    /// <summary>
    /// Nonce 长度（AES-GCM 推荐 12 字节）
    /// </summary>
    private const int NonceSize = 12;
    
    /// <summary>
    /// 认证标签长度（AES-GCM 固定 16 字节）
    /// </summary>
    private const int AuthTagSize = 16;
    
    /// <summary>
    /// MAC 长度（HMAC-SHA256 = 32 字节）
    /// </summary>
    private const int MacSize = 32;
    
    /// <summary>
    /// 包头长度（不加密）
    /// 长度(4B) + 类型(4B) = 8B
    /// </summary>
    private const int HeaderSize = 8;
    
    // ============================================================
    // 加密
    // ============================================================
    
    /// <summary>
    /// 加密消息
    /// </summary>
    /// <param name="sessionKey">会话密钥</param>
    /// <param name="messageId">消息类型</param>
    /// <param name="plaintext">明文数据</param>
    /// <returns>加密后的数据包</returns>
    public byte[] Encrypt(
        SessionKey sessionKey, 
        int messageId, 
        byte[] plaintext)
    {
        // ========================================================
        // 步骤1：生成随机 Nonce
        // ========================================================
        
        var nonce = new byte[NonceSize];
        RandomNumberGenerator.Fill(nonce);
        
        // ========================================================
        // 步骤2：创建 AuthTag 数组
        // ========================================================
        
        var cipherText = new byte[plaintext.Length];
        var authTag = new byte[AuthTagSize];
        
        // ========================================================
        // 步骤3：AES-GCM 加密
        // ========================================================
        
        using var aesGcm = new AesGcm(sessionKey.EncryptKey, AuthTagSize);
        aesGcm.Encrypt(
            nonce,
            plaintext,
            cipherText,
            authTag);
        
        // ========================================================
        // 步骤4：计算 HMAC
        // ========================================================
        
        // HMAC 覆盖：Nonce + CipherText + AuthTag
        var dataToMac = new byte[NonceSize + cipherText.Length + AuthTagSize];
        Buffer.BlockCopy(nonce, 0, dataToMac, 0, NonceSize);
        Buffer.BlockCopy(cipherText, 0, dataToMac, NonceSize, cipherText.Length);
        Buffer.BlockCopy(authTag, 0, dataToMac, NonceSize + cipherText.Length, AuthTagSize);
        
        using var hmac = new HMACSHA256(sessionKey.MacKey);
        var mac = hmac.ComputeHash(dataToMac);
        
        // ========================================================
        // 步骤5：组装最终数据包
        // ========================================================
        
        // 包格式：Nonce(12) + CipherText(N) + AuthTag(16) + MAC(32)
        var result = new byte[NonceSize + cipherText.Length + AuthTagSize + MacSize];
        
        Buffer.BlockCopy(nonce, 0, result, 0, NonceSize);
        Buffer.BlockCopy(cipherText, 0, result, NonceSize, cipherText.Length);
        Buffer.BlockCopy(authTag, 0, result, NonceSize + cipherText.Length, AuthTagSize);
        Buffer.BlockCopy(mac, 0, result, NonceSize + cipherText.Length + AuthTagSize, MacSize);
        
        return result;
    }
    
    // ============================================================
    // 解密
    // ============================================================
    
    /// <summary>
    /// 解密消息
    /// </summary>
    /// <param name="sessionKey">会话密钥</param>
    /// <param name="ciphertext">密文数据</param>
    /// <returns>解密后的消息类型和明文数据</returns>
    public (int messageId, byte[] plaintext) Decrypt(
        SessionKey sessionKey, 
        byte[] ciphertext)
    {
        // ========================================================
        // 步骤1：检查数据长度
        // ========================================================
        
        if (ciphertext.Length < NonceSize + AuthTagSize + MacSize + 4)
        {
            throw new SecurityException("加密数据长度不足");
        }
        
        // ========================================================
        // 步骤2：解析数据包
        // ========================================================
        
        var offset = 0;
        
        var nonce = new byte[NonceSize];
        Buffer.BlockCopy(ciphertext, offset, nonce, 0, NonceSize);
        offset += NonceSize;
        
        var cipherTextLength = ciphertext.Length - NonceSize - AuthTagSize - MacSize;
        var cipherText = new byte[cipherTextLength];
        Buffer.BlockCopy(ciphertext, offset, cipherText, 0, cipherTextLength);
        offset += cipherTextLength;
        
        var authTag = new byte[AuthTagSize];
        Buffer.BlockCopy(ciphertext, offset, authTag, 0, AuthTagSize);
        offset += AuthTagSize;
        
        var mac = new byte[MacSize];
        Buffer.BlockCopy(ciphertext, offset, mac, 0, MacSize);
        
        // ========================================================
        // 步骤3：验证 HMAC
        // ========================================================
        
        var dataToMac = new byte[ciphertext.Length - MacSize];
        Buffer.BlockCopy(ciphertext, 0, dataToMac, 0, dataToMac.Length);
        
        using var hmac = new HMACSHA256(sessionKey.MacKey);
        var computedMac = hmac.ComputeHash(dataToMac);
        
        if (!CryptographicOperations.FixedTimeEquals(mac, computedMac))
        {
            Logger.Warning("HMAC 验证失败，可能存在中间人攻击");
            throw new SecurityException("HMAC 验证失败");
        }
        
        // ========================================================
        // 步骤4：AES-GCM 解密
        // ========================================================
        
        var plaintext = new byte[cipherTextLength];
        
        using var aesGcm = new AesGcm(sessionKey.EncryptKey, AuthTagSize);
        aesGcm.Decrypt(
            nonce,
            cipherText,
            authTag,
            plaintext);
        
        return (0, plaintext); // messageId 从明文中解析
    }
}
```

### 3.4 防重放攻击

```csharp
/// <summary>
/// 防重放攻击管理器
/// 
/// 工作原理：
/// 1. 为每个消息分配递增的序列号
/// 2. 服务器记录最近 N 个已处理的序列号
/// 3. 收到重复序列号时拒绝处理
/// 4. 定期清理过期的序列号记录
/// </summary>
public class AntiReplayManager
{
    // ============================================================
    // 配置
    // ============================================================
    
    /// <summary>
    /// 序列号窗口大小
    /// 记录最近 1000 个序列号
    /// </summary>
    private const int WindowSize = 1000;
    
    /// <summary>
    /// 序列号过期时间（秒）
    /// 超过这个时间的序列号将被清理
    /// </summary>
    private const int ExpirySeconds = 300; // 5分钟
    
    // ============================================================
    // 数据结构
    // ============================================================
    
    /// <summary>
    /// 玩家序列号记录
    /// </summary>
    private class PlayerSequenceRecord
    {
        /// <summary>
        /// 已使用的序列号集合（滑动窗口）
        /// </summary>
        public readonly SortedSet<long> UsedSequences = new();
        
        /// <summary>
        /// 最后清理时间
        /// </summary>
        public DateTimeOffset LastCleanup { get; set; }
        
        /// <summary>
        /// 最后一个有效序列号
        /// </summary>
        public long LastValidSequence { get; set; }
    }
    
    /// <summary>
    /// 玩家记录字典
    /// </summary>
    private readonly ConcurrentDictionary<long, PlayerSequenceRecord> _records = new();
    
    // ============================================================
    // 核心方法
    // ============================================================
    
    /// <summary>
    /// 检查序列号是否有效
    /// 
    /// 返回值：
    /// - true: 序列号有效，可以处理
    /// - false: 序列号无效（重复或过期）
    /// </summary>
    /// <param name="playerId">玩家ID</param>
    /// <param name="sequenceNumber">序列号</param>
    /// <returns>是否有效</returns>
    public bool CheckSequence(long playerId, long sequenceNumber)
    {
        var record = _records.GetOrAdd(playerId, _ => new PlayerSequenceRecord
        {
            LastCleanup = DateTimeOffset.UtcNow
        });
        
        lock (record)
        {
            // ========================================================
            // 步骤1：清理过期序列号
            // ========================================================
            
            var now = DateTimeOffset.UtcNow;
            if (now - record.LastCleanup > TimeSpan.FromSeconds(ExpirySeconds))
            {
                CleanupExpiredSequences(record, now);
                record.LastCleanup = now;
            }
            
            // ========================================================
            // 步骤2：检查序列号是否过小（可能被重放）
            // ========================================================
            
            if (sequenceNumber < record.LastValidSequence - WindowSize)
            {
                Logger.Warning($"玩家 {playerId} 序列号过小: " +
                    $"序列号={sequenceNumber}, " +
                    $"最后有效={record.LastValidSequence}");
                return false;
            }
            
            // ========================================================
            // 步骤3：检查序列号是否重复
            // ========================================================
            
            if (record.UsedSequences.Contains(sequenceNumber))
            {
                Logger.Warning($"玩家 {playerId} 检测到重放攻击: " +
                    $"序列号={sequenceNumber} 已使用");
                return false;
            }
            
            // ========================================================
            // 步骤4：添加序列号到集合
            // ========================================================
            
            record.UsedSequences.Add(sequenceNumber);
            record.LastValidSequence = Math.Max(record.LastValidSequence, sequenceNumber);
            
            // 如果集合太大，移除最旧的
            while (record.UsedSequences.Count > WindowSize)
            {
                record.UsedSequences.Remove(record.UsedSequences.Min);
            }
            
            return true;
        }
    }
    
    /// <summary>
    /// 清理过期的序列号
    /// </summary>
    private void CleanupExpiredSequences(PlayerSequenceRecord record, DateTimeOffset now)
    {
        // 注意：这个实现依赖消息中的时间戳
        // 如果没有时间戳，可以使用序列号大小来判断
        
        var minValidSequence = record.LastValidSequence - WindowSize;
        
        // 移除小于最小有效序列号的记录
        var toRemove = record.UsedSequences
            .Where(s => s < minValidSequence)
            .ToList();
        
        foreach (var seq in toRemove)
        {
            record.UsedSequences.Remove(seq);
        }
    }
    
    /// <summary>
    /// 重置玩家的序列号记录（登出时）
    /// </summary>
    public void Reset(long playerId)
    {
        _records.TryRemove(playerId, out _);
```

### 4.1 连接安全

```csharp
/// <summary>
/// 连接安全管理器
/// 
/// 职责：
/// 1. IP 黑名单管理
/// 2. IP 白名单管理（可选）
/// 3. 单IP连接数限制
/// 4. 地理位置限制
/// 5. 连接速率限制
/// </summary>
public class ConnectionSecurityManager
{
    // ============================================================
    // 配置
    // ============================================================
    
    /// <summary>
    /// 单IP最大连接数
    /// </summary>
    private const int MaxConnectionsPerIp = 5;
    
    /// <summary>
    /// IP黑名单过期时间（小时）
    /// </summary>
    private const int BlacklistExpiryHours = 24;
    
    // ============================================================
    // 数据结构
    // ============================================================
    
    /// <summary>
    /// IP记录
    /// </summary>
    private class IpRecord
    {
        /// <summary>
        /// 连接数
        /// </summary>
        public int ConnectionCount;
        
        /// <summary>
        /// 最后连接时间
        /// </summary>
        public DateTimeOffset LastConnection;
        
        /// <summary>
        /// 连接尝试次数
        /// </summary>
        public int ConnectionAttempts;
        
        /// <summary>
        /// 最后尝试时间
        /// </summary>
        public DateTimeOffset LastAttempt;
    }
    
    /// <summary>
    /// IP记录字典
    /// </summary>
    private readonly ConcurrentDictionary<string, IpRecord> _ipRecords = new();
    
    /// <summary>
    /// IP黑名单
    /// </summary>
    private readonly ConcurrentDictionary<string, DateTimeOffset> _blacklist = new();
    
    /// <summary>
    /// 允许的国家/地区列表（可选）
    /// </summary>
    private readonly HashSet<string> _allowedCountries = new();
    
    // ============================================================
    // 核心方法
    // ============================================================
    
    /// <summary>
    /// 检查IP是否允许连接
    /// </summary>
    /// <param name="ipEndPoint">客户端IP</param>
    /// <returns>检查结果</returns>
    public ConnectionCheckResult CheckConnection(IPEndPoint ipEndPoint)
    {
        var ip = ipEndPoint.Address.ToString();
        
        // ========================================================
        // 步骤1：检查黑名单
        // ========================================================
        
        if (_blacklist.TryGetValue(ip, out var banExpiry))
        {
            if (DateTimeOffset.UtcNow < banExpiry)
            {
                Logger.Debug($"IP {ip} 在黑名单中，到期: {banExpiry}");
                return ConnectionCheckResult.Rejected("IP被封禁");
            }
            
            // 黑名单已过期，移除
            _blacklist.TryRemove(ip, out _);
        }
        
        // ========================================================
        // 步骤2：检查连接数限制
        // ========================================================
        
        var record = _ipRecords.GetOrAdd(ip, _ => new IpRecord());
        
        lock (record)
        {
            // 清理过期的连接记录（超过5分钟的连接不算）
            var now = DateTimeOffset.UtcNow;
            if (now - record.LastConnection > TimeSpan.FromMinutes(5))
            {
                record.ConnectionCount = 0;
            }
            
            if (record.ConnectionCount >= MaxConnectionsPerIp)
            {
                Logger.Warning($"IP {ip} 连接数超限: {record.ConnectionCount}");
                return ConnectionCheckResult.Rejected("连接数超限");
            }
        }
        
        // ========================================================
        // 步骤3：检查连接速率（防暴力破解）
        // ========================================================
        
        lock (record)
        {
            var attemptInterval = now - record.LastAttempt;
            
            // 如果1秒内尝试超过10次，限制
            if (attemptInterval < TimeSpan.FromSeconds(1))
            {
                record.ConnectionAttempts++;
                
                if (record.ConnectionAttempts > 10)
                {
                    Logger.Warning($"IP {ip} 连接频率异常: " +
                        $"{record.ConnectionAttempts}次/秒");
                    
                    // 临时封禁
                    _blacklist[ip] = DateTimeOffset.UtcNow.AddMinutes(5);
                    
                    return ConnectionCheckResult.Rejected("连接频率异常");
                }
            }
            else
            {
                record.ConnectionAttempts = 1;
            }
            
            record.LastAttempt = now;
        }
        
        return ConnectionCheckResult.Accepted;
    }
    
    /// <summary>
    /// 记录新连接
    /// </summary>
    public void RecordConnection(IPEndPoint ipEndPoint)
    {
        var ip = ipEndPoint.Address.ToString();
        
        var record = _ipRecords.GetOrAdd(ip, _ => new IpRecord());
        
        lock (record)
        {
            record.ConnectionCount++;
            record.LastConnection = DateTimeOffset.UtcNow;
        }
    }
    
    /// <summary>
    /// 移除连接记录
    /// </summary>
    public void RemoveConnection(IPEndPoint ipEndPoint)
    {
        var ip = ipEndPoint.Address.ToString();
        
        if (_ipRecords.TryGetValue(ip, out var record))
        {
            lock (record)
            {
                record.ConnectionCount = Math.Max(0, record.ConnectionCount - 1);
            }
        }
    }
    
    /// <summary>
    /// 将IP加入黑名单
    /// </summary>
    public void BanIp(string ip, string reason, TimeSpan duration)
    {
        var expiry = DateTimeOffset.UtcNow.Add(duration);
        _blacklist[ip] = expiry;
        
        Logger.Warning($"IP {ip} 被加入黑名单: {reason}, 到期: {expiry}");
        
        // TODO: 断开该IP的所有连接
    }
    
    /// <summary>
    /// 将IP从黑名单移除
    /// </summary>
    public void UnbanIp(string ip)
    {
        _blacklist.TryRemove(ip, out _);
        Logger.Info($"IP {ip} 已从黑名单移除");
    }
}

/// <summary>
/// 连接检查结果
/// </summary>
public class ConnectionCheckResult
{
    public bool IsAccepted { get; private set; }
    public string Reason { get; private set; }
    
    public static ConnectionCheckResult Accepted => 
        new() { IsAccepted = true };
    
    public static ConnectionCheckResult Rejected(string reason) =>
```

### 4.2 数据包验证

```csharp
/// <summary>
/// 数据包验证器
/// 
/// 职责：
/// 1. 验证数据包大小是否合理
/// 2. 验证消息类型是否有效
/// 3. 验证消息内容格式
/// 4. 检测异常数据包
/// </summary>
public class PacketValidator
{
    // ============================================================
    // 配置
    // ============================================================
    
    /// <summary>
    /// 单个消息最大大小（字节）
    /// 1MB
    /// </summary>
    private const int MaxMessageSize = 1024 * 1024;
    
    /// <summary>
    /// 最小消息大小（字节）
    /// </summary>
    private const int MinMessageSize = 1;
    
    /// <summary>
    /// 允许的消息类型范围
    /// </summary>
    private const int MinMessageType = 0x0000;
    private const int MaxMessageType = 0xFFFF;
    
    // ============================================================
    // 验证方法
    // ============================================================
    
    /// <summary>
    /// 验证消息大小
    /// </summary>
    public bool ValidateMessageSize(int size)
    {
        if (size < MinMessageSize || size > MaxMessageSize)
        {
            Logger.Warning($"消息大小异常: {size} 字节 " +
                $"(范围: {MinMessageSize}-{MaxMessageSize})");
            return false;
        }
        
        return true;
    }
    
    /// <summary>
    /// 验证消息类型
    /// </summary>
    public bool ValidateMessageType(int messageType)
    {
        if (messageType < MinMessageType || messageType > MaxMessageType)
        {
            Logger.Warning($"消息类型异常: 0x{messageType:X4} " +
                $"(范围: 0x{MinMessageType:X4}-0x{MaxMessageType:X4})");
            return false;
        }
        
        return true;
    }
    
    /// <summary>
    /// 验证消息内容是否包含非法字符
    /// </summary>
    public bool ValidateMessageContent(byte[] content)
    {
        // 检查是否包含空字节（可能表示二进制注入）
        for (int i = 0; i < Math.Min(100, content.Length); i++)
        {
            if (content[i] == 0x00)
            {
                // 空字节在某些位置是合法的，这里只做警告
                Logger.Debug($"消息内容包含空字节: 位置={i}");
            }
        }
        
        // 检查字符串内容（如果有的话）
        // 防止 XSS、SQL注入等
        if (IsProbablyText(content))
        {
            var text = Encoding.UTF8.GetString(content);
            return !ContainsInjection(text);
        }
        
        return true;
    }
    
    /// <summary>
    /// 检测是否为文本内容
    /// </summary>
    private bool IsProbablyText(byte[] data)
    {
        // 简单启发式：如果大部分字节是可打印字符，认为是文本
        int printableCount = 0;
        foreach (var b in data.Take(100))
        {
            if ((b >= 0x20 && b <= 0x7E) || b == 0x09 || b == 0x0A || b == 0x0D)
            {
                printableCount++;
            }
        }
        
        return printableCount > data.Length * 0.7;
    }
    
    /// <summary>
    /// 检测注入攻击
    /// </summary>
    private bool ContainsInjection(string text)
    {
        // SQL注入关键词
        var sqlKeywords = new[] { "SELECT", "INSERT", "UPDATE", "DELETE", 
            "DROP", "UNION", "--", ";", "/*", "*/" };
        
        foreach (var keyword in sqlKeywords)
        {
            if (text.Contains(keyword, StringComparison.OrdinalIgnoreCase))
            {
                Logger.Warning($"检测到SQL注入关键词: {keyword}");
                return true;
            }
        }
        
        // XSS关键词
        var xssKeywords = new[] { "<script", "javascript:", "onerror=", "onclick=" };
        
        foreach (var keyword in xssKeywords)
        {
            if (text.Contains(keyword, StringComparison.OrdinalIgnoreCase))
            {
                Logger.Warning($"检测到XSS关键词: {keyword}");
                return true;
            }
        }
        
        return false;
    }
}
```

### 5.2 密码安全

```csharp
/// <summary>
/// 密码安全服务
/// 
/// 安全措施：
/// 1. 密码加盐哈希（PBKDF2 / Argon2）
/// 2. 密码强度验证
/// 3. 登录失败锁定
/// 4. 登录历史记录
/// </summary>
public class PasswordSecurityService
{
    // ============================================================
    // 配置
    // ============================================================
    
    /// <summary>
    /// 哈希算法迭代次数（PBKDF2）
    /// </summary>
    private const int Pbkdf2Iterations = 100000;
    
    /// <summary>
    /// 盐值长度（字节）
    /// </summary>
    private const int SaltSize = 32;
    
    /// <summary>
    /// 哈希长度（字节）
    /// </summary>
    private const int HashSize = 32;
    
    /// <summary>
    /// 最大登录失败次数
    /// </summary>
    private const int MaxLoginFailures = 5;
    
    /// <summary>
    /// 登录失败锁定时间
    /// </summary>
    private static readonly TimeSpan LockoutDuration = TimeSpan.FromMinutes(30);
    
    // ============================================================
    // 密码哈希
    // ============================================================
    
    /// <summary>
    /// 哈希密码
    /// 
    /// 使用 PBKDF2-HMAC-SHA256 进行安全哈希
    /// </summary>
    /// <param name="password">明文密码</param>
    /// <returns>哈希结果（Base64编码的盐值:哈希值）</returns>
    public string HashPassword(string password)
    {
        // 生成随机盐值
        var salt = new byte[SaltSize];
        RandomNumberGenerator.Fill(salt);
        
        // 计算哈希
        using var pbkdf2 = new Rfc2898DeriveBytes(
            password, 
            salt, 
            Pbkdf2Iterations, 
            HashAlgorithmName.SHA256);
        
        var hash = pbkdf2.GetBytes(HashSize);
        
        // 组合盐值和哈希
        var combined = new byte[SaltSize + HashSize];
        Buffer.BlockCopy(salt, 0, combined, 0, SaltSize);
        Buffer.BlockCopy(hash, 0, combined, SaltSize, HashSize);
        
        // 返回 Base64 编码
        return $"{Convert.ToBase64String(salt)}:{Convert.ToBase64String(hash)}";
    }
    
    /// <summary>
    /// 验证密码
    /// </summary>
    /// <param name="password">待验证的密码</param>
    /// <param name="storedHash">存储的哈希（格式：盐值:哈希值）</param>
    /// <returns>是否匹配</returns>
    public bool VerifyPassword(string password, string storedHash)
    {
        var parts = storedHash.Split(':');
        if (parts.Length != 2)
        {
            Logger.Warning("密码哈希格式错误");
            return false;
        }
        
        var salt = Convert.FromBase64String(parts[0]);
        var storedHashBytes = Convert.FromBase64String(parts[1]);
        
        // 使用相同的盐值计算哈希
        using var pbkdf2 = new Rfc2898DeriveBytes(
            password, 
            salt, 
            Pbkdf2Iterations, 
            HashAlgorithmName.SHA256);
        
        var computedHash = pbkdf2.GetBytes(HashSize);
        
        // 使用恒定时间比较
        return CryptographicOperations.FixedTimeEquals(computedHash, storedHashBytes);
    }
    
    /// <summary>
    /// 检查密码强度
    /// </summary>
    public PasswordStrength CheckPasswordStrength(string password)
    {
        var strength = PasswordStrength.Weak;
        var score = 0;
        
        // 长度检查
        if (password.Length >= 8) score++;
        if (password.Length >= 12) score++;
        if (password.Length >= 16) score++;
        
        // 字符类型检查
        if (password.Any(char.IsLower)) score++;
        if (password.Any(char.IsUpper)) score++;
        if (password.Any(char.IsDigit)) score++;
        if (password.Any(c => !char.IsLetterOrDigit(c))) score++;
        
        // 常见密码检查
        var commonPasswords = new[] { "password", "123456", "qwerty", "admin" };
        if (commonPasswords.Any(p => password.Contains(p, StringComparison.OrdinalIgnoreCase)))
        {
            score = 0;
        }
        
        // 计算强度
        if (score >= 6) strength = PasswordStrength.Strong;
        else if (score >= 4) strength = PasswordStrength.Medium;
        else if (score >= 2) strength = PasswordStrength.Weak;
        else strength = PasswordStrength.VeryWeak;
        
        return strength;
    }
    
    // ============================================================
    // 登录失败锁定
    // ============================================================
    
    /// <summary>
    /// 登录失败记录
    /// </summary>
    private class LoginFailureRecord
    {
        public int FailureCount { get; set; }
        public DateTimeOffset? LockoutEnd { get; set; }
        public DateTimeOffset LastFailure { get; set; }
    }
    
    /// <summary>
    /// 登录失败记录字典
    /// </summary>
    private readonly ConcurrentDictionary<string, LoginFailureRecord> _loginFailures = new();
    
    /// <summary>
    /// 检查账号是否被锁定
    /// </summary>
    public bool IsAccountLocked(string account)
    {
        if (_loginFailures.TryGetValue(account, out var record))
        {
            if (record.LockoutEnd.HasValue && 
                DateTimeOffset.UtcNow < record.LockoutEnd.Value)
            {
                return true;
            }
        }
        
        return false;
    }
    
    /// <summary>
    /// 记录登录失败
    /// </summary>
    public void RecordLoginFailure(string account)
    {
        var record = _loginFailures.GetOrAdd(account, _ => new LoginFailureRecord());
        
        lock (record)
        {
            record.FailureCount++;
            record.LastFailure = DateTimeOffset.UtcNow;
            
            // 如果失败次数超过阈值，锁定账号
            if (record.FailureCount >= MaxLoginFailures)
            {
                record.LockoutEnd = DateTimeOffset.UtcNow.Add(LockoutDuration);
                
                Logger.Warning($"账号 {account} 登录失败次数过多，已锁定: " +
                    $"失败次数={record.FailureCount}, " +
                    $"锁定结束={record.LockoutEnd}");
            }
        }
    }
    
    /// <summary>
    /// 清除登录失败记录（登录成功时调用）
    /// </summary>
    public void ClearLoginFailure(string account)
    {
        _loginFailures.TryRemove(account, out _);
    }
}

/// <summary>
/// 密码强度枚举
/// </summary>
public enum PasswordStrength
{
    VeryWeak,
    Weak,
    Medium,
    Strong
}
```

## 8. 消息限流 (v2 新增)

### 8.1 攻击场景

| 场景 | 描述 | 影响 |
|------|------|------|
| 高频连接 | 单个 IP 每秒建立上万连接 | 耗尽端口/内存/CPU |
| 高频消息 | 单个玩家每秒发送数千条消息 | 耗尽消息队列 |
| 恶意放大 | 构造超大消息体 | 耗尽带宽/内存 |

### 8.2 令牌桶算法原理

```
        ┌──────────────────────────────────────┐
        │  令牌桶 (maxTokens = 10000)          │
        │                                      │
        │   ◆ ◆ ◆ ◆ ◆ ◆ ◆ ◆ ◆ ◆               │
        │   ◆ ◆ ◆ ◆ ◆ ◆                       │
        │   ◆ ◆ ◆                              │ ← 消耗 1 令牌 = 处理 1 条消息
        │                                      │
        └──────▲───────────────────────────────┘
               │
               │ 补充令牌 (refillAmount = 10000 / 1s)
               │
        ┌──────┴──────┐
        │  定时器     │ 每 intervalMs 毫秒补充一次
        └─────────────┘
```

### 8.3 RateLimiter 实现

```csharp
public class RateLimiter
{
    private long _tokens;
    private long _lastRefillTime;
    private readonly long _refillAmount;
    private readonly long _maxTokens;
    private readonly long _intervalMs;

    public RateLimiter(long refillAmount, long maxTokens, long intervalMs = 1000)
    {
        _refillAmount = refillAmount;
        _maxTokens = maxTokens;
        _intervalMs = intervalMs;
        _tokens = maxTokens;
        _lastRefillTime = DateTime.UtcNow.Ticks;
    }

    public bool TryConsume(long amount = 1)
    {
        // 1. 先补充令牌（懒惰补充，避免定时锁）
        long now = DateTime.UtcNow.Ticks;
        long elapsedMs = (now - _lastRefillTime) / TimeSpan.TicksPerMillisecond;
        if (elapsedMs >= _intervalMs)
        {
            long refillCount = elapsedMs / _intervalMs;
            long newTokens = Interlocked.Add(ref _tokens, refillCount * _refillAmount);
            if (newTokens > _maxTokens) Interlocked.Exchange(ref _tokens, _maxTokens);
            Interlocked.Exchange(ref _lastRefillTime, now);
        }

        // 2. 尝试消耗令牌
        long current = Interlocked.Read(ref _tokens);
        if (current < amount) return false;
        Interlocked.Add(ref _tokens, -amount);
        return true;
    }
}
```

### 8.4 两级限流配置

```csharp
// 会话级：单玩家 100 TPS，最大突发 200 令牌
var sessionLimiter = new RateLimiter(100, 200, 1000);

// 全局级：整服 10000 TPS，最大突发 50000 令牌
var globalLimiter = new RateLimiter(10000, 50000, 1000);

// 使用顺序：先全局后会话，避免无效消耗
if (globalLimiter.TryConsume() && sessionLimiter.TryConsume())
{
    await messageRouter.RouteAsync(session, messageId, data);
}
else
{
    Logger.Warning("Security", "消息被限流丢弃，会话: {0}", session.SessionId);
}
```

### 8.5 限流指标监控

```csharp
// 在消息路由层集成指标监控
if (!globalLimiter.TryConsume())
{
    MetricsCollector.Instance.IncrementCounter("server.rate_limited_global", 1);
}
```

## 9. 服务状态管理 (v2 新增)

### 9.1 状态流转

```
  ┌─────────┐     Start()      ┌──────────┐    Running()   ┌──────────┐
  │ Stopped │ ───────────────► │ Starting │ ─────────────► │ Running  │
  └─────────┘                  └──────────┘                └────┬─────┘
       ▲                                                         │
       │ Stop()                                            Pause()│
       │                                                         ▼
  ┌────┴──────┐     Stop()     ┌─────────┐    Pausing()    ┌──────────┐
  │ Stopping  │ ◄───────────── │ Running │ ◄─────────────  │ Pausing  │
  └───────────┘                └─────────┘                └────┬─────┘
                                                                 │
                                                          Resume()│
                                                                 ▼
                                                          ┌──────────┐
                                                          │  Paused  │
                                                          └──────────┘
```

### 9.2 使用示例

```csharp
// 初始化状态
ServiceStateManager.Instance.SetStarting();
ServiceStateManager.Instance.StateChanged += (s, e) => 
    Logger.Info("Server", "状态变更: {0} -> {1}", e.Previous, e.Current);

// 启动完成
ServiceStateManager.Instance.SetRunning();

// 优雅关闭
ServiceStateManager.Instance.SetStopping();
await Task.Delay(5000); // 等待处理完消息
ServiceStateManager.Instance.SetStopped();
```
