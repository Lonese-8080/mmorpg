## ⚠️ 实现状态说明

### 已实现部分

- ✅ **固定帧率调度（`GameScheduler`）**：默认 20 FPS，可配置目标帧率
- ✅ **FPS 统计与帧时间监控**：最近 100 帧的滑动平均，超时告警
- ✅ **帧事件机制（`OnFrameStart` → `OnUpdate` → `OnFrameEnd`）**：通过 `FrameEventArgs` 传递帧信息
- ✅ **消息队列（`MessageChannel<T>`）**：支持多生产者/单消费者，核心消息分发通道

### 待开发部分（规划中）

- ⏳ **消息优先级队列**：按优先级处理消息，高优先级（心跳/战斗）优先，低优先级（聊天）延后
- ⏳ **拥塞控制与背压通知**：队列长度监控，超出阈值通知客户端减速，丢弃低优先级消息
- ⏳ **ECS 系统调度**：分帧执行系统，按帧预算管理系统执行数量
- ⏳ **复杂负载均衡策略**：跨线程/进程调度，动态分配计算资源

> **说明**：第 1 节和第 4 节的基础帧调度机制已实现。第 2、3 节的消息优先级与拥塞控制为设计方案，后续将逐步实现。

### 2.2 优先级消息队列

```csharp
/// <summary>
/// 优先级消息队列
/// 
/// 注意：此类为设计示例，当前代码库中不存在此类
/// 
/// 使用多级队列实现优先级：
/// - 每一帧先处理高优先级队列
/// - 处理不完的延后到下一帧
/// - 低优先级队列只在系统空闲时处理
/// </summary>
public class PriorityMessageQueue
{
    // ============================================================
    // 配置
    // ============================================================
    
    /// <summary>
    /// 每个优先级的最大队列长度
    /// </summary>
    private static readonly Dictionary<MessagePriority, int> MaxQueueLengths = new()
    {
        [MessagePriority.Critical] = 1000,
        [MessagePriority.High] = 5000,
        [MessagePriority.Normal] = 10000,
        [MessagePriority.Low] = 20000,
        [MessagePriority.Background] = 50000
    };
    
    // ============================================================
    // 队列
    // ============================================================
    
    /// <summary>
    /// 优先级队列数组
    /// </summary>
    private readonly Channel<QueuedMessage>[] _queues;
    
    /// <summary>
    /// 队列统计
    /// </summary>
    private readonly Counter<long>[] _queueLengths;
    
    // ============================================================
    // 初始化
    // ============================================================
    
    public PriorityMessageQueue()
    {
        // 为每个优先级创建队列
        _queues = new Channel<QueuedMessage>[5];
        _queueLengths = new Counter<long>[5];
        
        for (int i = 0; i < 5; i++)
        {
            var priority = (MessagePriority)i;
            var maxLength = MaxQueueLengths[priority];
            
            _queues[i] = Channel.CreateBounded<QueuedMessage>(
                new BoundedChannelOptions(maxLength)
                {
                    SingleReader = true,
                    SingleWriter = false,
                    FullMode = BoundedChannelFullMode.Wait
                });
            
            _queueLengths[i] = new Counter<long>($"queue_{priority}");
        }
    }
    
    // ============================================================
    // 入队
    // ============================================================
    
    /// <summary>
    /// 入队
    /// </summary>
    public async Task EnqueueAsync(QueuedMessage message)
    {
        var priority = message.Priority;
        var queueIndex = (int)priority;
        
        // 检查队列长度
        if (_queues[queueIndex].Reader.Count >= MaxQueueLengths[priority])
        {
            // 队列满了，根据优先级决定策略
            if (priority >= MessagePriority.Low)
            {
                // 低优先级：丢弃
                Logger.Warning("MessageQueue", 
                    "队列满，丢弃低优先级消息: Type={0}, Priority={1}",
                    message.MessageType, priority);
                return;
            }
            
            // 高优先级：等待（阻塞）
            // 这可能会导致帧超时，但为了保证关键消息不丢失
            Logger.Warning("MessageQueue",
                "高优先级队列满，等待空间: Priority={0}",
                priority);
        }
        
        await _queues[queueIndex].Writer.WriteAsync(message);
        _queueLengths[queueIndex].Increment();
    }
    
    // ============================================================
    // 出队（按优先级）
    // ============================================================
    
    /// <summary>
    /// 按优先级出队
    /// 先取高优先级，直到处理完或时间用完
    /// </summary>
    public async Task<QueuedMessage?> DequeueAsync(TimeSpan timeout)
    {
        var deadline = DateTimeOffset.UtcNow.Add(timeout);
        
        // 按优先级顺序尝试获取
        for (int i = 0; i < 5; i++)
        {
            if (_queues[i].Reader.TryRead(out var message))
            {
                _queueLengths[i].Decrement();
                return message;
            }
            
            // 如果还有时间，尝试下一个优先级
            if (DateTimeOffset.UtcNow >= deadline)
                break;
        }
        
        return null;
    }
    
    /// <summary>
    /// 批量出队
    /// 在指定时间内尽可能多地取出消息
    /// </summary>
    public List<QueuedMessage> DequeueBatch(TimeSpan budget)
    {
        var result = new List<QueuedMessage>();
        var deadline = DateTimeOffset.UtcNow.Add(budget);
        
        while (DateTimeOffset.UtcNow < deadline)
        {
            var message = DequeueAsync(TimeSpan.FromMilliseconds(1)).Result;
            if (message != null)
            {
                result.Add(message);
            }
            else
            {
                break; // 没有更多消息
            }
        }
        
        return result;
    }
    
    // ============================================================
    // 统计
    // ============================================================
    
    /// <summary>
    /// 获取队列长度
    /// </summary>
    public int GetQueueLength(MessagePriority priority)
    {
        return _queues[(int)priority].Reader.Count;
    }
    
    /// <summary>
    /// 获取所有队列长度
    /// </summary>
    public Dictionary<MessagePriority, int> GetAllQueueLengths()
    {
        var result = new Dictionary<MessagePriority, int>();
        
        for (int i = 0; i < 5; i++)
        {
            result[(MessagePriority)i] = _queues[i].Reader.Count;
        }
        
        return result;
    }
}

/// <summary>
/// 队列消息
/// 
/// 注意：此为设计示例类，当前代码库中不存在
/// </summary>
public class QueuedMessage
{
    /// <summary>
    /// 会话
    /// </summary>
    public Session Session { get; set; }
    
    /// <summary>
    /// 消息类型
    /// </summary>
    public int MessageType { get; set; }
    
    /// <summary>
    /// 消息数据
    /// </summary>
    public byte[] Data { get; set; }
    
    /// <summary>
    /// 优先级
    /// </summary>
    public MessagePriority Priority { get; set; }
    
    /// <summary>
    /// 入队时间
    /// </summary>

}

/// <summary>
/// 计数器
/// 
/// 注意：此为设计示例类，当前代码库中不存在
/// </summary>
public class Counter<T>
{
    private long _value;
    
    public Counter(string name) { }
    
    public void Increment() => Interlocked.Increment(ref _value);
    public void Decrement() => Interlocked.Decrement(ref _value);
    public long Value => _value;
}
```

## 3. 拥塞控制机制（规划中：CongestionControlManager、背压通知）

### 3.1 拥塞检测

```csharp
/// <summary>
/// 拥塞控制管理器
/// 
/// 注意：此类为设计示例，当前代码库中不存在此类
/// 
/// 职责：
/// 1. 监控消息处理延迟
/// 2. 检测队列积压
/// 3. 触发背压机制
/// 4. 动态调整处理策略
/// </summary>
public class CongestionControlManager
{
    // ============================================================
    // 配置
    // ============================================================
    
    /// <summary>
    /// 警告队列长度
    /// </summary>
    private const int WarningQueueLength = 1000;
    
    /// <summary>
    /// 严重队列长度
    /// </summary>
    private const int CriticalQueueLength = 5000;
    
    /// <summary>
    /// 警告处理延迟（毫秒）
    /// </summary>
    private const int WarningLatencyMs = 50;
    
    /// <summary>
    /// 严重处理延迟（毫秒）
    /// </summary>
    private const int CriticalLatencyMs = 200;
    
    // ============================================================
    // 状态
    // ============================================================
    
    /// <summary>
    /// 当前拥塞级别
    /// </summary>
    private CongestionLevel _currentLevel = CongestionLevel.Normal;
    
    /// <summary>
    /// 最后拥塞级别变化时间
    /// </summary>
    private DateTimeOffset _lastLevelChange;
    
    /// <summary>
    /// 处理延迟统计
    /// </summary>
    private readonly RingBuffer<long> _latencyBuffer = new(1000);
    
    /// <summary>
    /// 背压会话列表
    /// </summary>
    private readonly HashSet<long> _backpressureSessions = new();
    
    // ============================================================
    // 事件
    // ============================================================
    
    public event Action<CongestionLevel> OnLevelChanged;
    
    // ============================================================
    // 更新
    // ============================================================
    
    /// <summary>
    /// 记录处理延迟
    /// </summary>
    public void RecordLatency(long latencyMs)
    {
        _latencyBuffer.Add(latencyMs);
    }
    
    /// <summary>
    /// 检查拥塞状态
    /// </summary>
    public void CheckCongestion(PriorityMessageQueue queue)
    {
        // 获取统计
        var queueLengths = queue.GetAllQueueLengths();
        var totalQueueLength = queueLengths.Values.Sum();
        var avgLatency = _latencyBuffer.Average();
        
        // 计算新的拥塞级别
        var newLevel = CalculateLevel(totalQueueLength, avgLatency);
        
        if (newLevel != _currentLevel)
        {
            ChangeLevel(newLevel);
        }
        
        // 应用背压
        ApplyBackpressure(queue, newLevel);
        
        // 记录指标
        MetricsCollector.Instance.RecordQueueLength(totalQueueLength);
        
        // 记录告警
        if (newLevel >= CongestionLevel.Warning)
        {
            MetricsCollector.Instance.RecordError("Congestion");
            
            Logger.Warning("Congestion", 
                "拥塞检测: Level={0}, QueueLength={1}, Latency={2}ms",
                newLevel, totalQueueLength, avgLatency);
        }
    }
    
    /// <summary>
    /// 计算拥塞级别
    /// </summary>
    private CongestionLevel CalculateLevel(int queueLength, double avgLatency)
    {
        // 严重拥塞
        if (queueLength >= CriticalQueueLength || avgLatency >= CriticalLatencyMs)
            return CongestionLevel.Critical;
        
        // 警告拥塞
        if (queueLength >= WarningQueueLength || avgLatency >= WarningLatencyMs)
            return CongestionLevel.Warning;
        
        // 轻微拥塞
        if (queueLength >= WarningQueueLength / 2 || avgLatency >= WarningLatencyMs / 2)
            return CongestionLevel.Degraded;
        
        return CongestionLevel.Normal;
    }
    
    /// <summary>
    /// 变更拥塞级别
    /// </summary>
    private void ChangeLevel(CongestionLevel newLevel)
    {
        var oldLevel = _currentLevel;
        _currentLevel = newLevel;
        _lastLevelChange = DateTimeOffset.UtcNow;
        
        Logger.Info("Congestion", 
            "拥塞级别变更: {0} -> {1}",
            oldLevel, newLevel);
        
        OnLevelChanged?.Invoke(newLevel);
    }
    
    /// <summary>
    /// 应用背压机制
    /// </summary>
    private void ApplyBackpressure(PriorityMessageQueue queue, CongestionLevel level)
    {
        // 根据拥塞级别限制某些操作
        
        switch (level)
        {
            case CongestionLevel.Critical:
                // 严重拥塞：只处理 Critical 和 High 优先级
                // 向所有客户端发送背压通知
                BroadcastBackpressureNotification(true);
                break;
                
            case CongestionLevel.Warning:
                // 警告拥塞：处理到 Normal 优先级
                BroadcastBackpressureNotification(false);
                break;
                
            default:
                // 正常：处理所有优先级
                if (_backpressureSessions.Count > 0)
                {
                    _backpressureSessions.Clear();
                    BroadcastBackpressureNotification(false);
                }
                break;
        }
    }
    
    /// <summary>
    /// 广播背压通知
    /// </summary>
    private void BroadcastBackpressureNotification(bool severe)
    {
        var notification = new S2C_BackpressureNotification
        {
            Enabled = severe,
            Message = severe 
                ? "服务器繁忙，部分功能暂时受限" 
                : "服务器已恢复正常"
        };
        
        Server.BroadcastAsync(notification);
    }
}

/// <summary>
/// 拥塞级别
/// 
/// 注意：此为设计示例枚举，当前代码库中不存在
/// </summary>
public enum CongestionLevel
{
    /// <summary>
    /// 正常
    /// </summary>
    Normal,
    
    /// <summary>
    /// 轻微拥塞
    /// </summary>
    Degraded,
    
    /// <summary>
    /// 警告拥塞
    /// </summary>
    Warning,
    
    /// <summary>
    /// 严重拥塞
    /// </summary>
    Critical
}

/// <summary>
/// 环形缓冲区
/// 
/// 注意：此为设计示例类，当前代码库中不存在此类
/// </summary>
public class RingBuffer<T>
{
    private readonly T[] _buffer;
    private int _index;
    private long _count;
    
    public RingBuffer(int capacity)
    {
        _buffer = new T[capacity];
    }
    
    public void Add(T value)
    {
        _buffer[_index] = value;
        _index = (_index + 1) % _buffer.Length;
        _count++;
    }
    
    public double Average()
    {
        if (_count == 0) return 0;
        
        dynamic sum = 0;
        var count = (int)Math.Min(_count, _buffer.Length);
        
        for (int i = 0; i < count; i++)
        {
            sum += _buffer[i];
        }
        
        return sum / count;
    }
}
```

```csharp
/// <summary>

///

/// </summary>
public class FrameBudgetAllocation
{
    /// <summary>
    /// 消息处理预算（百分比）
    /// </summary>
    public double MessageProcessingPercent { get; set; } = 0.40;
    
    /// <summary>
    /// ECS系统执行预算（百分比）
    /// </summary>
    public double EcsSystemsPercent { get; set; } = 0.30;
    
    /// <summary>
    /// 状态同步预算（百分比）
    /// </summary>
    public double StateSyncPercent { get; set; } = 0.15;
    
    /// <summary>
    /// 预留预算（百分比）
    /// </summary>
    public double ReservedPercent { get; set; } = 0.15;
    
    /// <summary>
    /// 计算预算（毫秒）
    /// </summary>
    public double GetBudgetMs(double frameBudgetMs, double percent)
    {
        return frameBudgetMs * percent;
    }
}
```

### 4.2 帧预算管理实现

```csharp
/// <summary>
/// 帧预算管理器
/// 
/// 注意：此类为设计示例类。实际代码库中由 `GameScheduler` 实现帧调度和 FPS 统计
/// 
/// 职责：
/// 1. 跟踪每帧的时间使用
/// 2. 防止单帧超时
/// 3. 动态调整任务分配
/// 4. 记录帧率统计
/// </summary>
public class FrameBudgetManager
{
    // ============================================================
    // 配置
    // ============================================================
    
    private readonly GameLoopConfig _config;
    
    // ============================================================
    // 状态
    // ============================================================
    
    /// <summary>
    /// 帧开始时间
    /// </summary>
    private long _frameStartTime;
    
    /// <summary>
    /// 当前帧已用时间
    /// </summary>
    private double _currentFrameUsed;
    
    /// <summary>
    /// 帧号
    /// </summary>
    private long _frameNumber;
    
    /// <summary>
    /// 帧率统计
    /// </summary>
    private readonly RingBuffer<double> _fpsBuffer = new(100);
    
    /// <summary>
    /// 每帧时间统计
    /// </summary>
    private readonly RingBuffer<double> _frameTimeBuffer = new(100);
    
    // ============================================================
    // 初始化
    // ============================================================
    
    public FrameBudgetManager(GameLoopConfig config)
    {
        _config = config;
    }
    
    // ============================================================
    // 帧控制
    // ============================================================
    
    /// <summary>
    /// 开始新帧
    /// </summary>
    public void BeginFrame()
    {
        _frameStartTime = Stopwatch.GetTimestamp();
        _frameNumber++;
    }
    
    /// <summary>
    /// 结束帧
    /// </summary>
    public void EndFrame()
    {
        var frameEndTime = Stopwatch.GetTimestamp();
        var frameTimeMs = (frameEndTime - _frameStartTime) * 1000.0 / Stopwatch.Frequency;
        
        _currentFrameUsed = frameTimeMs;
        _frameTimeBuffer.Add(frameTimeMs);
        
        var fps = 1000.0 / frameTimeMs;
        _fpsBuffer.Add(fps);
        
        // 记录指标
        MetricsCollector.Instance.RecordFrameRate(fps, 
            frameTimeMs / _config.FrameBudgetMs);
        
        // 检查是否超时
        if (frameTimeMs > _config.MaxFrameTimeMs)
        {
            Logger.Warning("GameLoop", 
                "帧超时: Frame={0}, Time={1}ms, Budget={2}ms",
                _frameNumber, frameTimeMs, _config.FrameBudgetMs);
        }
        
        // 等待下一帧（如果需要）
        WaitForNextFrame(frameTimeMs);
    }
    
    /// <summary>
    /// 等待下一帧
    /// </summary>
    private void WaitForNextFrame(double frameTimeMs)
    {
        var sleepTime = _config.FrameBudgetMs - frameTimeMs;
        
        if (sleepTime > 0)
        {
            Thread.Sleep((int)sleepTime);
        }
    }
    
    // ============================================================
    // 预算控制
    // ============================================================
    
    /// <summary>
    /// 检查预算是否足够
    /// </summary>
    public bool HasBudget(double budgetMs)
    {
        var elapsed = GetElapsedMs();
        var remaining = _config.FrameBudgetMs - elapsed;
        return remaining >= budgetMs;
    }
    
    /// <summary>
    /// 获取已用时间
    /// </summary>
    public double GetElapsedMs()
    {
        var currentTime = Stopwatch.GetTimestamp();
        return (currentTime - _frameStartTime) * 1000.0 / Stopwatch.Frequency;
    }
    
    /// <summary>
    /// 获取剩余预算
    /// </summary>
    public double GetRemainingBudget()
    {
        return Math.Max(0, _config.FrameBudgetMs - GetElapsedMs());
    }
    
    /// <summary>
    /// 执行任务并检查预算
    /// </summary>
    public async Task<bool> ExecuteWithBudgetAsync(Func<Task> task, 
        double budgetMs, string taskName)
    {
        var startTime = Stopwatch.GetTimestamp();
        
        // 检查预算
        if (!HasBudget(budgetMs))
        {
            Logger.Debug("GameLoop", 
                "预算不足，跳过任务: {0}, 剩余={1}ms",
                taskName, GetRemainingBudget());
            return false;
        }
        
        try
        {
            await task();
            
            var elapsed = (Stopwatch.GetTimestamp() - startTime) * 1000.0 / Stopwatch.Frequency;
            
            if (elapsed > budgetMs)
            {
                Logger.Warning("GameLoop",
                    "任务超出预算: {0}, 耗时={1}ms, 预算={2}ms",
                    taskName, elapsed, budgetMs);
            }
            
            return true;
        }
        catch (Exception ex)
        {
            Logger.Error("GameLoop", ex,
                "任务执行失败: {0}", taskName);
            return false;
        }
    }
    
    // ============================================================
    // 统计
    // ============================================================
    
    /// <summary>
    /// 获取平均帧率
    /// </summary>
    public double GetAverageFps()
    {
        return _fpsBuffer.Average() > 0 ? 1000.0 / _fpsBuffer.Average() : 0;
    }
    
    /// <summary>
    /// 获取平均帧时间
    /// </summary>
    public double GetAverageFrameTime()
    {
        return _frameTimeBuffer.Average();
    }
    
    /// <summary>
    /// 获取帧预算使用率
    /// </summary>
    public double GetBudgetUsage()
    {
        return _currentFrameUsed / _config.FrameBudgetMs;
    }
}
```
