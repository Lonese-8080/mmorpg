## 二、使用 Vagrant 快速创建 VM（一键启动）

### 安装 Vagrant

```bash
# Windows: 下载安装 https://developer.hashicorp.com/vagrant/downloads
# macOS: brew install vagrant
# Linux: sudo apt install vagrant
```

### 创建 Vagrantfile

```bash
# 在项目根目录创建
cat > Vagrantfile << 'EOF'
# -*- mode: ruby -*-
# vi: set ft=ruby :

Vagrant.configure("2") do |config|
  # ==================== 一体化 VM（开发验证用）====================
  config.vm.define "mmorpg-allinone" do |node|
    node.vm.box = "ubuntu/jammy64"
    node.vm.hostname = "mmorpg-allinone"
    node.vm.network "private_network", ip: "192.168.56.10"
    node.vm.network "forwarded_port", guest: 7001, host: 7001
    node.vm.network "forwarded_port", guest: 7002, host: 7002
    node.vm.network "forwarded_port", guest: 8080, host: 8080
    node.vm.network "forwarded_port", guest: 3000, host: 3000
    node.vm.network "forwarded_port", guest: 9090, host: 9090
    node.vm.network "forwarded_port", guest: 16686, host: 16686
    node.vm.provider "virtualbox" do |vb|
      vb.memory = "8192"
      vb.cpus = 4
      vb.linked_clone = true
      vb.customize ["modifyvm", :id, "--natdnshostresolver1", "on"]
    end
    node.vm.provision "shell", path: "scripts/vm-setup.sh"
  end

  # ==================== 游戏服务器 VM ========================
  config.vm.define "mmorpg-server" do |node|
    node.vm.box = "ubuntu/jammy64"
    node.vm.hostname = "mmorpg-server"
    node.vm.network "private_network", ip: "192.168.56.11"
    node.vm.network "forwarded_port", guest: 7001, host: 7001
    node.vm.network "forwarded_port", guest: 8080, host: 8080
    node.vm.provider "virtualbox" do |vb|
      vb.memory = "8192"
      vb.cpus = 4
      vb.linked_clone = true
    end
    node.vm.provision "shell", path: "scripts/vm-setup.sh"
  end
end
EOF
```

### 启动 VM

```bash
# 创建并启动所有 VM
vagrant up

# 只启动特定 VM
vagrant up mmorpg-allinone

# SSH 登录
vagrant ssh mmorpg-allinone
```

## 三、VM 初始化脚本（自动安装所有依赖）

创建 `scripts/vm-setup.sh`：

```bash
#!/bin/bash
set -e

echo "========== MMORPG Framework VM Setup =========="

# 更新系统
echo "[1/8] 更新系统..."
sudo apt update && sudo apt upgrade -y

# 安装基础工具
echo "[2/8] 安装基础工具..."
sudo apt install -y curl wget git unzip vim htop net-tools \
    ca-certificates gnupg lsb-release software-properties-common

# 安装 Docker
echo "[3/8] 安装 Docker..."
if ! command -v docker &> /dev/null; then
    curl -fsSL https://get.docker.com | sh
    sudo usermod -aG docker vagrant
fi

# 安装 Docker Compose
echo "[4/8] 安装 Docker Compose..."
sudo curl -L "https://github.com/docker/compose/releases/download/v2.24.0/docker-compose-$(uname -s)-$(uname -m)" \
    -o /usr/local/bin/docker-compose
sudo chmod +x /usr/local/bin/docker-compose
docker-compose --version

# 安装 .NET 10 SDK
echo "[5/8] 安装 .NET 10 SDK..."
if ! command -v dotnet &> /dev/null; then
    wget https://dot.net/v1/dotnet-install.sh -O dotnet-install.sh
    chmod +x dotnet-install.sh
    sudo ./dotnet-install.sh --channel 10.0 --install-dir /usr/share/dotnet
    sudo ln -s /usr/share/dotnet/dotnet /usr/bin/dotnet
fi
dotnet --version

# 安装压力测试工具
echo "[6/8] 安装压测工具..."
sudo apt install -y apache2-utils  # 包含 ab
if ! command -v bombardier &> /dev/null; then
    wget https://github.com/rakyll/bombardier/releases/download/v1.11.0/bombardier_linux_amd64 -O /tmp/bombardier
    sudo mv /tmp/bombardier /usr/local/bin/bombardier
    sudo chmod +x /usr/local/bin/bombardier
fi

# 安装 Python（用于压测脚本）
echo "[7/8] 安装 Python..."
sudo apt install -y python3 python3-pip python3-venv

# 防火墙配置（开发环境关闭，生产环境请按需配置）
echo "[8/8] 配置防火墙（开发模式）..."
sudo ufw disable || true

echo ""
echo "========== Setup Complete! =========="
echo "请运行 'vagrant reload' 以加载 Docker 组权限"
echo "然后运行 'cd /vagrant && docker-compose up -d' 启动所有服务"
```

## 四、一体化 Docker Compose 配置（所有服务）

创建项目根目录 `docker-compose.yml`：

```yaml
version: '3.8'

services:
  # ==================== 游戏服务器 ====================
  gameserver:
    build:
      context: .
      dockerfile: Dockerfile
    container_name: mmorpg-gameserver
    image: mmorpg/framework:latest
    ports:
      - "7001:7001"   # 游戏端口1
      - "7002:7002"   # 游戏端口2
      - "7003:7003"   # 游戏端口3
      - "8080:8080"   # HTTP 管理端口
    environment:
      - ASPNETCORE_ENVIRONMENT=Production
      - OTEL_EXPORTER_OTLP_ENDPOINT=http://jaeger:4317
      - PROMETHEUS__PORT=9090
    volumes:
      - ./config/appsettings.json:/app/config/appsettings.json:ro
      - ./config/resilience.json:/app/config/resilience.json:ro
    depends_on:
      - redis
      - prometheus
      - jaeger
    networks:
      - mmorpg-net
    restart: unless-stopped
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8080/health"]
      interval: 30s
      timeout: 10s
      retries: 3
      start_period: 40s

  # ==================== Redis（缓存/配置中心） ====================
  redis:
    image: redis:7-alpine
    container_name: mmorpg-redis
    ports:
      - "6379:6379"
    volumes:
      - redis-data:/data
    command: redis-server --appendonly yes --maxmemory 512mb --maxmemory-policy allkeys-lru
    networks:
      - mmorpg-net
    restart: unless-stopped
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s
      timeout: 5s
      retries: 3

  # ==================== Prometheus（指标收集） ====================
  prometheus:
    image: prom/prometheus:v2.48.0
    container_name: mmorpg-prometheus
    ports:
      - "9090:9090"
    volumes:
      - ./config/prometheus.yml:/etc/prometheus/prometheus.yml:ro
      - prometheus-data:/prometheus
    command:
      - '--config.file=/etc/prometheus/prometheus.yml'
      - '--storage.tsdb.path=/prometheus'
      - '--storage.tsdb.retention.time=15d'
      - '--web.enable-lifecycle'
    networks:
      - mmorpg-net
    restart: unless-stopped

  # ==================== Grafana（监控大盘） ====================
  grafana:
    image: grafana/grafana:10.2.2
    container_name: mmorpg-grafana
    ports:
      - "3000:3000"
    environment:
      - GF_SECURITY_ADMIN_USER=admin
      - GF_SECURITY_ADMIN_PASSWORD=admin123
      - GF_USERS_ALLOW_SIGN_UP=false
      - GF_SERVER_ROOT_URL=http://localhost:3000
    volumes:
      - grafana-data:/var/lib/grafana
      - ./config/grafana/provisioning:/etc/grafana/provisioning:ro
    depends_on:
      - prometheus
    networks:
      - mmorpg-net
    restart: unless-stopped

  # ==================== Jaeger（分布式追踪） ====================
  jaeger:
    image: jaegertracing/all-in-one:1.53
    container_name: mmorpg-jaeger
    ports:
      - "16686:16686"  # UI
      - "6831:6831"    # UDP 接收器
      - "4317:4317"    # gRPC 接收器（OTLP）
      - "4318:4318"    # HTTP 接收器（OTLP）
    environment:
      - COLLECTOR_OTLP_ENABLED=true
    volumes:
      - jaeger-data:/tmp/jaeger
    networks:
      - mmorpg-net
    restart: unless-stopped

  # ==================== AlertManager（告警） ====================
  alertmanager:
    image: prom/alertmanager:v0.26.0
    container_name: mmorpg-alertmanager
    ports:
      - "9093:9093"
    volumes:
      - ./config/alertmanager.yml:/etc/alertmanager/alertmanager.yml:ro
      - alertmanager-data:/alertmanager
    networks:
      - mmorpg-net
    restart: unless-stopped

networks:
  mmorpg-net:
    driver: bridge

volumes:
  redis-data:
  prometheus-data:
  grafana-data:
  jaeger-data:
  alertmanager-data:
```

## 四、Dockerfile 多阶段构建与非 root 容器

> **重要**：生产镜像必须使用非 root 用户运行，防止容器逃逸后获得宿主机 root 权限。

### 4.1 多阶段 Dockerfile

```dockerfile
# ===========================
# 第一阶段：SDK 构建镜像
# ===========================
FROM mcr.microsoft.com/dotnet/sdk:10.0 AS builder

WORKDIR /src

# 复制解决方案文件和项目文件
COPY ["MMORPG.sln", "MMORPG.sln"]
COPY ["src/MMORPG.Framework/MMORPG.Framework.csproj", "src/MMORPG.Framework/"]
COPY ["src/MMORPG.Server/MMORPG.Server.csproj", "src/MMORPG.Server/"]

# 还原依赖（利用 Docker 层缓存）
RUN dotnet restore MMORPG.sln

# 复制源码
COPY . .

# 发布（输出到 /app/publish）
RUN dotnet publish src/MMORPG.Server/MMORPG.Server.csproj \
    -c Release \
    -o /app/publish \
    --no-restore

# ===========================
# 第二阶段：ASP.NET 运行时镜像（精简）
# ===========================
FROM mcr.microsoft.com/dotnet/aspnet:10.0 AS runtime

# ---------- 安全：创建非 root 用户 ----------
# 如果 GID 1000 已被占用，使用 10000 避免冲突
RUN groupadd -g 10000 appgroup || true
RUN useradd -m -u 10000 -g appgroup appuser

WORKDIR /app

# 复制发布产物（从 builder 阶段）
COPY --from=builder /app/publish .

# ---------- 安全：切换到非 root ----------
USER appuser

# 健康检查（使用非特权端口）
HEALTHCHECK --interval=30s --timeout=10s --start-period=5s --retries=3 \
    CMD curl -f http://localhost:8080/health || exit 1

# 默认启动命令
ENTRYPOINT ["dotnet", "MMORPG.Server.dll"]
```

### 4.2 构建命令

```bash
# 构建 SDK 镜像（调试用，包含完整 .NET SDK）
docker build -t mmorpg-server:dev --target builder -f Dockerfile .

# 构建生产镜像（最小运行时，--target runtime）
docker build -t mmorpg-server:prod --target runtime -f Dockerfile .

# 查看镜像大小
docker images mmorpg-server:prod
# REPOSITORY   TAG     SIZE
# mmorpg-server prod    208MB  ← 仅运行时镜像

# 查看构建阶段
docker build --target builder -t mmorpg-server:builder -f Dockerfile . && \
docker images mmorpg-server:builder
# REPOSITORY   TAG     SIZE
# mmorpg-server builder  1.2GB  ← SDK 镜像很大，仅构建用
```

### 4.3 GID 冲突问题（常见错误）

如果构建时报错：
```
groupadd: GID '1000' already exists
```

解决：修改 Dockerfile 中的 GID 为 10000：

```dockerfile
RUN groupadd -g 10000 appgroup || true
RUN useradd -m -u 10000 -g appgroup appuser
```

### 4.4 在 docker-compose.yml 中使用

```yaml
services:
  gameserver:
    # 使用多阶段构建
    build:
      context: .
      dockerfile: Dockerfile
      target: runtime  # ← 指定运行时阶段（不用 builder）
    image: mmorpg/framework:latest  # 构建后 tag 的名称
    user: "10000:10000"  # 显式指定用户（双重保险）
    ports:
      - "7001:7001"
      - "8080:8080"
      - "9091:9091"
    environment:
      - DOTNET_RUNNING_IN_CONTAINER=true
      - ASPNETCORE_URLS=http://+:8080
    restart: unless-stopped
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8080/health"]
      interval: 30s
      timeout: 10s
      retries: 3
```

### 4.5 非 root 容器的安全价值

| 场景 | root 用户容器 | 非 root 容器（appuser） |
|------|-------------|----------------------|
| 攻击者入侵容器 | 可直接修改系统文件、安装工具 | 受 UID/GID 10000 限制，无法修改系统文件 |
| 挂载敏感目录 | 可直接读写 | 受文件权限限制（如果 volume 是 root 创建） |
| 横向移动 | 可能通过 Docker Socket | 无法操作 Docker Socket |
| 特权模式逃逸 | 可能获得宿主机 root | 即使逃逸也受用户命名空间限制 |

**最佳实践**：始终使用非 root 用户运行应用容器，服务端口（非 80/443）使用非特权端口（如 8080 而非 80）。

## 五、配置文件

### Prometheus 配置 `config/prometheus.yml`

```yaml
global:
  scrape_interval: 15s
  evaluation_interval: 15s

alerting:
  alertmanagers:
    - static_configs:
        - targets:
            - alertmanager:9093

rule_files:
  - "alerts.yml"

scrape_configs:
  # 游戏服务器指标
  - job_name: 'mmorpg-gameserver'
    static_configs:
      - targets: ['gameserver:8080']
    metrics_path: '/metrics'

  # Redis 指标
  - job_name: 'redis'
    static_configs:
      - targets: ['redis:6379']

  # Prometheus 自身
  - job_name: 'prometheus'
    static_configs:
      - targets: ['localhost:9090']
```

### Grafana 数据源 Provisioning `config/grafana/provisioning/datasources/prometheus.yml`

```yaml
apiVersion: 1

datasources:
  - name: Prometheus
    type: prometheus
    access: proxy
    url: http://prometheus:9090
    isDefault: true
    editable: false

  - name: Jaeger
    type: jaeger
    access: proxy
    url: http://jaeger:16686
    editable: false
```

### Grafana Dashboard Provisioning `config/grafana/provisioning/dashboards/dashboard.yml`

```yaml
apiVersion: 1

providers:
  - name: 'MMORPG Framework'
    orgId: 1
    folder: 'MMORPG'
    folderUid: ''
    type: file
    disableDeletion: false
    updateIntervalSeconds: 30
    allowUiUpdates: true
    options:
      path: /etc/grafana/provisioning/dashboards
```

### AlertManager 配置 `config/alertmanager.yml`

```yaml
global:
  resolve_timeout: 5m

route:
  group_by: ['alertname']
  group_wait: 10s
  group_interval: 10s
  repeat_interval: 1h
  receiver: 'email'
  routes:
    - match:
        severity: critical
      receiver: 'email'

receivers:
  - name: 'email'
    email_configs:
      - to: 'admin@example.com'
        from: 'alertmanager@example.com'
        smarthost: 'smtp.example.com:587'
        auth_username: 'alertmanager@example.com'
        auth_password: 'changeme'

inhibit_rules:
  - source_match:
      severity: 'critical'
    target_match:
      severity: 'warning'
    equal: ['alertname']
```
