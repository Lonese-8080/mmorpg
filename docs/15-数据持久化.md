## 2. 存档策略

### 2.1 存档时机

```csharp
/// <summary>
/// 存档策略
/// 
/// 核心原则：
/// 1. 关键操作立即存档（避免数据丢失）
/// 2. 定期自动存档（确保数据完整性）
/// 3. 玩家下线时存档（保存最终状态）
/// 4. 服务器关闭时存档（保护在线玩家数据）
/// </summary>
public class SaveStrategy
{
    // ============================================================
    // 配置
    // ============================================================
    
    /// <summary>
    /// 自动存档间隔（秒）
    /// </summary>
    private const int AutoSaveIntervalSeconds = 300; // 5分钟
    
    /// <summary>
    /// 离线玩家保留时间（秒）
    /// </summary>
    private const int OfflineRetentionSeconds = 300; // 5分钟
    
    // ============================================================
    // 存档时机
    // ============================================================
    
    /// <summary>
    /// 关键操作存档（立即保存）
    /// </summary>
    public void OnCriticalOperation(Player player, string operation)
    {
        Logger.Info("Save", 
            "关键操作存档: PlayerId={0}, Operation={1}",
            player.Id, operation);
        
        // 立即保存
        _ = SavePlayerAsync(player);
    }
    
    /// <summary>
    /// 定期存档（定时任务）
    /// </summary>
    public void OnPeriodicSave()
    {
        Logger.Debug("Save", "执行定期存档");
        
        foreach (var player in Server.GetOnlinePlayers())
        {
            // 只保存变化的玩家
            if (player.HasUnsavedChanges)
            {
                _ = SavePlayerAsync(player);
            }
        }
    }
    
    /// <summary>
    /// 玩家下线存档
    /// </summary>
    public async Task OnPlayerLogout(Player player)
    {
        Logger.Info("Save", 
            "玩家下线存档: PlayerId={0}, OnlineTime={1}分钟",
            player.Id, player.GetOnlineMinutes());
        
        // 确保保存完成
        await SavePlayerAsync(player);
        
        // 设置离线保留
        await SetOfflineRetentionAsync(player);
    }
    
    /// <summary>
    /// 服务器关闭存档
    /// </summary>
    public async Task OnServerShutdown()
    {
        Logger.Info("Save", "服务器关闭，保存所有玩家数据");
        
        var tasks = new List<Task>();
        
        foreach (var player in Server.GetOnlinePlayers())
        {
            tasks.Add(SavePlayerAsync(player));
        }
        
        await Task.WhenAll(tasks);
        
        Logger.Info("Save", "所有玩家数据已保存");
    }
}
```
