## ⚠️ 实现状态说明

### 已实现部分

- ✅ **心跳检测（`HeartbeatManager`）**：定期检查 Session 心跳，超时触发断线
- ✅ **Session 断开逻辑（`Session.DisconnectAsync`）**：正常关闭连接，触发断线事件
- ✅ **连接状态跟踪（`DisconnectDetector` 概念逻辑）**：消息接收计数、最后心跳时间记录

### 待开发部分（规划中）

- ⏳ **重连 Token 验证机制**：断线时生成 Token，客户端重连时验证
- ⏳ **玩家状态保存与恢复（`SessionSnapshot` / `PlayerStateManager`）**：断线前保存状态快照，重连时恢复
- ⏳ **离线期间消息处理（`ReconnectManager`）**：离线期间重要消息缓存，重连后同步
- ⏳ **登录状态保持与自动重连**：客户端支持快速重连恢复游戏状态
- ⏳ **断线后的游戏世界状态同步**：确保重连时玩家与世界状态保持一致

> **说明**：本文档详细描述了完整断线重连机制的设计。第 1、2 节涉及的心跳检测与连接断开已实现。第 3、4、5 节为设计方案，后续将逐步实现。

## 2. 断线检测机制（已实现：心跳检测 + Session 断开）

### 2.1 多层次检测

```csharp
/// <summary>
/// 断线检测管理器
/// 
/// 注意：此为设计示例类，当前代码库中以 `HeartbeatManager` 和 `Session.DisconnectAsync` 的形式实现部分功能
/// 
/// 多层次检测机制：
/// 1. 底层 Socket 检测
/// 2. 心跳检测
/// 3. 应用层状态检测
/// </summary>
public class DisconnectDetector
{
    // ============================================================
    // 字段
    // ============================================================
    
    /// <summary>
    /// 玩家会话状态
    /// </summary>
    private readonly ConcurrentDictionary<long, PlayerConnectionState> _states = new();
    
    /// <summary>
    /// 心跳管理器
    /// </summary>
    private readonly HeartbeatManager _heartbeatManager;
    
    /// <summary>
    /// 事件回调
    /// </summary>
    
    /// <summary>
    /// 断线回调
    /// Key: PlayerId
    /// Value: 断线时间和原因
    /// </summary>
    public event Action<long, DisconnectReason, DateTimeOffset> OnDisconnected;
    
    /// <summary>
    /// 重连回调
    /// </summary>
    public event Action<long, DateTimeOffset, TimeSpan> OnReconnected;
    
    // ============================================================
    // 状态管理
    // ============================================================
    
    /// <summary>
    /// 记录玩家连接
    /// </summary>
    public void RecordConnection(long playerId)
    {
        var state = new PlayerConnectionState
        {
            PlayerId = playerId,
            ConnectedAt = DateTimeOffset.UtcNow,
            LastHeartbeat = DateTimeOffset.UtcNow,
            LastMessageTime = DateTimeOffset.UtcNow,
            DisconnectTimeout = GetDisconnectTimeout()
        };
        
        _states[playerId] = state;
        
        Logger.Debug("Reconnect", "玩家 {0} 连接已记录，IP超时: {1}", playerId, state.DisconnectTimeout);
    }
    
    /// <summary>
    /// 记录玩家心跳
    /// </summary>
    public void RecordHeartbeat(long playerId)
    {
        if (_states.TryGetValue(playerId, out var state))
        {
            state.LastHeartbeat = DateTimeOffset.UtcNow;
            state.HeartbeatCount++;
        }
    }
    
    /// <summary>
    /// 记录玩家消息
    /// </summary>
    public void RecordMessage(long playerId)
    {
        if (_states.TryGetValue(playerId, out var state))
        {
            state.LastMessageTime = DateTimeOffset.UtcNow;
            state.MessageCount++;
        }
    }
    
    /// <summary>
    /// 记录玩家断线
    /// </summary>
    public void RecordDisconnection(long playerId, DisconnectReason reason)
    {
        if (_states.TryGetValue(playerId, out var state))
        {
            state.DisconnectedAt = DateTimeOffset.UtcNow;
            state.DisconnectReason = reason;
            
            Logger.Info("Reconnect", "玩家 {0} 断线: 原因={1}, 在线时长={2}, 心跳次数={3}, 消息数={4}",
                playerId, reason, state.GetOnlineDuration(), state.HeartbeatCount, state.MessageCount);
            
            OnDisconnected?.Invoke(playerId, reason, state.DisconnectedAt.Value);
        }
    }
    
    /// <summary>
    /// 获取玩家状态
    /// </summary>
    public PlayerConnectionState? GetState(long playerId)
    {
        _states.TryGetValue(playerId, out var state);
        return state;
    }
    
    /// <summary>
    /// 移除玩家状态（登出时）
    /// </summary>
    public void RemoveState(long playerId)
    {
        _states.TryRemove(playerId, out _);
    }
    
    // ============================================================
    // 检测逻辑
    // ============================================================
    
    /// <summary>
    /// 执行断线检测
    /// 每秒执行一次
    /// </summary>
    public void DetectDisconnections()
    {
        var now = DateTimeOffset.UtcNow;
        
        foreach (var (playerId, state) in _states)
        {
            // 跳过已断线的
            if (state.DisconnectedAt.HasValue)
                continue;
            
            // 检查心跳超时
            var heartbeatTimeout = now - state.LastHeartbeat;
            if (heartbeatTimeout > state.DisconnectTimeout)
            {
                Logger.Warning("Reconnect", "玩家 {0} 心跳超时: 最后心跳={1:F1}秒前, 超时阈值={2:F1}秒",
                    playerId, heartbeatTimeout.TotalSeconds, state.DisconnectTimeout.TotalSeconds);
                
                RecordDisconnection(playerId, DisconnectReason.HeartbeatTimeout);
                continue;
            }
            
            // 检查消息超时（可选，更宽松）
            var messageTimeout = now - state.LastMessageTime;
            if (messageTimeout > state.DisconnectTimeout * 2)
            {
                Logger.Warning("Reconnect", "玩家 {0} 消息超时: 最后消息={1:F1}秒前", playerId, messageTimeout.TotalSeconds);
            }
        }
    }
    
    /// <summary>
    /// 获取断线超时时间
    /// </summary>
    private TimeSpan GetDisconnectTimeout()
    {
        // 通常是心跳间隔的 3-5 倍
        // 例如：心跳间隔 5 秒 -> 超时 20 秒
        return TimeSpan.FromSeconds(20);
    }
}

/// <summary>
/// 玩家连接状态
/// 
/// 注意：此为设计示例类，当前代码库中不存在此类
/// </summary>
public class PlayerConnectionState
{
    public long PlayerId { get; set; }
    public DateTimeOffset ConnectedAt { get; set; }
    public DateTimeOffset? LastHeartbeat { get; set; }
    public DateTimeOffset? LastMessageTime { get; set; }
    public DateTimeOffset? DisconnectedAt { get; set; }
    public DisconnectReason? DisconnectReason { get; set; }
    public TimeSpan DisconnectTimeout { get; set; }
    public int HeartbeatCount { get; set; }
    public int MessageCount { get; set; }
    
    /// <summary>
    /// 获取在线时长
    /// </summary>
    public TimeSpan GetOnlineDuration()
    {
        var endTime = DisconnectedAt ?? DateTimeOffset.UtcNow;
        return endTime - ConnectedAt;
    }
}

/// <summary>
/// 断线原因
/// 
/// 注意：此为设计示例枚举，当前代码库中不存在
/// </summary>
public enum DisconnectReason
{
    ClientDisconnect,     // 客户端主动断开
    HeartbeatTimeout,    // 心跳超时
    NetworkError,        // 网络错误
    Kicked,              // 被踢出
    ServerShutdown,       // 服务器关闭
    SessionHijacked,     // 会话被顶替
    SecurityViolation,    // 安全违规
    Maintenance          // 维护
}
```

### 2.2 心跳超时处理

```csharp
/// <summary>
/// 心跳管理器
/// 
/// 注意：此类已在代码库中实现（`MMORPG.Core.HeartbeatManager`）
/// 
/// 职责：
/// 1. 发送心跳请求
/// 2. 检测心跳超时
/// 3. 触发断线处理
/// </summary>
public class HeartbeatManager
{
    // ============================================================
    // 配置
    // ============================================================
    
    /// <summary>
    /// 心跳间隔
    /// </summary>
    private const int HeartbeatIntervalSeconds = 5;
    
    /// <summary>
    /// 心跳超时次数
    /// 连续 N 次心跳超时才判定为断线
    /// </summary>
    private const int MaxMissedHeartbeats = 4;
    
    // ============================================================
    // 字段
    // ============================================================
    
    /// <summary>
    /// 玩家心跳状态
    /// </summary>
    private class PlayerHeartbeatState
    {
        public long LastHeartbeatAck { get; set; }
        public int MissedCount { get; set; }
        public int SentCount { get; set; }
    }
    
    private readonly ConcurrentDictionary<long, PlayerHeartbeatState> _heartbeats = new();
    
    /// <summary>
    /// 定时器
    /// </summary>
    private readonly Timer _timer;
    
    /// <summary>
    /// 回调
    /// </summary>
    public event Action<long> OnHeartbeatTimeout;
    
    // ============================================================
    // 初始化
    // ============================================================
    
    public HeartbeatManager()
    {
        _timer = new Timer(SendHeartbeats, null, 
            TimeSpan.Zero, 
            TimeSpan.FromSeconds(HeartbeatIntervalSeconds));
    }
    
    // ============================================================
    // 心跳处理
    // ============================================================
    
    /// <summary>
    /// 发送心跳
    /// </summary>
    private void SendHeartbeats(object state)
    {
        foreach (var session in Server.GetActiveSessions())
        {
            var playerId = session.PlayerId;
            
            // 记录发送
            var heartbeatState = _heartbeats.GetOrAdd(playerId, 
                _ => new PlayerHeartbeatState());
            
            heartbeatState.SentCount++;
            
            // 发送心跳请求
            session.SendAsync(new S2C_Heartbeat
            {
                ServerTime = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds()
            });
        }
    }
    
    /// <summary>
    /// 处理心跳响应
    /// </summary>
    public void HandleHeartbeatAck(Session session, C2S_HeartbeatAck ack)
    {
        var playerId = session.PlayerId;
        
        if (_heartbeats.TryGetValue(playerId, out var heartbeatState))
        {
            heartbeatState.LastHeartbeatAck = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds();
            heartbeatState.MissedCount = 0;
            
            // 更新连接状态
            DisconnectDetector.RecordHeartbeat(playerId);
        }
    }
    
    /// <summary>
    /// 检查心跳超时
    /// </summary>
    public void CheckHeartbeatTimeouts()
    {
        var now = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds();
        
        foreach (var (playerId, state) in _heartbeats)
        {
            var elapsed = now - state.LastHeartbeatAck;
            var timeout = HeartbeatIntervalSeconds * MaxMissedHeartbeats * 1000;
            
            if (elapsed > timeout)
            {
                Logger.Warning($"玩家 {playerId} 连续心跳超时，触发断线检测");
                
                OnHeartbeatTimeout?.Invoke(playerId);
                
                // 重置状态，允许重连
                state.MissedCount = 0;
            }
        }
    }
    
    /// <summary>
    /// 清理玩家心跳状态
    /// </summary>
    public void RemovePlayer(long playerId)
    {
        _heartbeats.TryRemove(playerId, out _);
    }
}
```

### 3.3 重连处理实现

```csharp
/// <summary>
/// 重连管理器
/// 
/// 注意：此类为设计示例，当前代码库中不存在此类
/// 
/// 职责：
/// 1. 管理重连 Token
/// 2. 验证重连请求
/// 3. 执行状态恢复
/// </summary>
public class ReconnectManager
{
    // ============================================================
    // 配置
    // ============================================================
    
    /// <summary>
    /// 重连保留期（秒）
    /// 断线后服务器保留玩家状态的时间
    /// </summary>
    private const int ReconnectRetentionSeconds = 300; // 5分钟
    
    /// <summary>
    /// 重连 Token 有效期（秒）
    /// </summary>
    private const int TokenValiditySeconds = 600; // 10分钟
    
    // ============================================================
    // 字段
    // ============================================================
    
    /// <summary>
    /// Redis 连接
    /// </summary>
    private readonly IRedis _redis;
    
    /// <summary>
    /// 玩家状态管理器
    /// </summary>
    private readonly PlayerStateManager _stateManager;
    
    // ============================================================
    // 公共方法
    // ============================================================
    
    /// <summary>
    /// 处理断线（生成重连 Token）
    /// </summary>
    /// <param name="playerId">玩家ID</param>
    /// <param name="sessionId">旧Session ID</param>
    /// <returns>重连 Token</returns>
    public async Task<string> HandleDisconnect(long playerId, long sessionId)
    {
        // 生成重连 Token
        var token = GenerateReconnectToken(playerId, sessionId);
        
        // 保存到 Redis
        var key = GetReconnectKey(playerId);
        var data = new ReconnectData
        {
            PlayerId = playerId,
            OldSessionId = sessionId,
            Token = token,
            DisconnectTime = DateTimeOffset.UtcNow,
            ExpiresAt = DateTimeOffset.UtcNow.AddSeconds(TokenValiditySeconds)
        };
        
        await _redis.SetAsync(key, JsonSerializer.Serialize(data), 
            TimeSpan.FromSeconds(TokenValiditySeconds));
        
        // 延长玩家状态保留期
        await _stateManager.ExtendRetention(playerId, 
            TimeSpan.FromSeconds(ReconnectRetentionSeconds));
        
        Logger.Info($"玩家 {playerId} 断线，已生成重连 Token: {token}");
        
        return token;
    }
    
    /// <summary>
    /// 处理重连请求
    /// </summary>
    public async Task<ReconnectResult> HandleReconnect(Session session, 
        C2S_Reconnect request)
    {
        var now = DateTimeOffset.UtcNow;
        
        // ========================================================
        // 步骤1：验证 Token
        // ========================================================
        
        var key = GetReconnectKey(request.PlayerId);
        var dataStr = await _redis.GetAsync(key);
        
        if (string.IsNullOrEmpty(dataStr))
        {
            Logger.Warning($"玩家 {request.PlayerId} 重连失败: Token 不存在或已过期");
            
            return new ReconnectResult
            {
                Success = false,
                ErrorCode = ReconnectErrorCode.SessionExpired,
                ErrorMessage = "重连已过期，请重新登录"
            };
        }
        
        var data = JsonSerializer.Deserialize<ReconnectData>(dataStr);
        
        if (data == null || data.Token != request.ReconnectToken)
        {
            Logger.Warning($"玩家 {request.PlayerId} 重连失败: Token 验证失败");
            
            return new ReconnectResult
            {
                Success = false,
                ErrorCode = ReconnectErrorCode.InvalidToken,
                ErrorMessage = "重连验证失败"
            };
        }
        
        // ========================================================
        // 步骤2：检查是否已被顶号
        // ========================================================
        
        var currentSession = Server.GetSession(request.PlayerId);
        if (currentSession != null && currentSession.Id != data.OldSessionId)
        {
            // 旧 Session 还没被清理，说明被顶号了
            Logger.Warning($"玩家 {request.PlayerId} 重连失败: 已被顶号");
            
            // 断开新的连接（让后来的赢）
            await currentSession.DisconnectAsync("账号在其他地方登录");
            
            return new ReconnectResult
            {
                Success = false,
                ErrorCode = ReconnectErrorCode.AlreadyOnline,
                ErrorMessage = "账号在其他地方登录"
            };
        }
        
        // ========================================================
        // 步骤3：恢复玩家状态
        // ========================================================
        
        var state = await _stateManager.RestoreState(request.PlayerId);
        
        if (state == null)
        {
            Logger.Warning($"玩家 {request.PlayerId} 重连失败: 状态恢复失败");
            
            return new ReconnectResult
            {
                Success = false,
                ErrorCode = ReconnectErrorCode.PlayerNotFound,
                ErrorMessage = "玩家数据加载失败"
            };
        }
        
        // ========================================================
        // 步骤4：绑定新 Session
        // ========================================================
        
        session.BindPlayer(request.PlayerId);
        
        // 更新 Redis 中的 Session 信息
        await _redis.SetAsync($"player:session:{request.PlayerId}",
            session.Id.ToString());
        
        // 清除重连 Token
        await _redis.DelAsync(key);
        
        // ========================================================
        // 步骤5：发送同步数据
        // ========================================================
        
        var disconnectDuration = (long)(now - data.DisconnectTime).TotalSeconds;
        
        // 计算离线期间错过的内容
        var missedEvents = await CalculateMissedEvents(request.PlayerId, 
            data.DisconnectTime);
        
        // 发送重连成功响应
        await session.SendAsync(new S2C_ReconnectResult
        {
            Success = true,
            DisconnectDuration = disconnectDuration,
            ServerTimestamp = now.ToUnixTimeMilliseconds(),
            NewSessionId = session.Id
        });
        
        // 发送离线期间错过的消息
        await SendMissedEvents(session, missedEvents);
        
        Logger.Info($"玩家 {request.PlayerId} 重连成功: " +
            $"断线时长={disconnectDuration}秒, " +
            $"错过事件={missedEvents.Count}个");
        
        return new ReconnectResult
        {
            Success = true,
            PlayerState = state,
            DisconnectDuration = disconnectDuration
        };
    }
    
    // ============================================================
    // 辅助方法
    // ============================================================
    
    /// <summary>
    /// 生成重连 Token
    /// </summary>
    private string GenerateReconnectToken(long playerId, long sessionId)
    {
        var data = $"{playerId}:{sessionId}:{DateTimeOffset.UtcNow.Ticks}";
        var hash = SHA256.HashData(Encoding.UTF8.GetBytes(data));
        return Convert.ToBase64String(hash)[..32];
    }
    
    /// <summary>
    /// 获取 Redis Key
    /// </summary>
    private string GetReconnectKey(long playerId)
    {
        return $"player:reconnect:{playerId}";
    }
    
    /// <summary>
    /// 计算离线期间错过的内容
    /// </summary>
    private async Task<List<MissedEvent>> CalculateMissedEvents(
        long playerId, 
        DateTimeOffset since)
    {
        var events = new List<MissedEvent>();
        
        // 获取离线期间的重要事件
        // 例如：聊天消息、系统公告、场景变化等
        
        // 1. 场景内玩家变化
        var nearbyPlayers = await SceneManager.GetNearbyPlayersAsync(
            playerId, since);
        foreach (var player in nearbyPlayers)
        {
            events.Add(new MissedEvent
            {
                Type = MissedEventType.PlayerEnter,
                Data = player
            });
        }
        
        // 2. 聊天消息
        var chats = await ChatManager.GetMissedMessagesAsync(
            playerId, since);
        foreach (var chat in chats)
        {
            events.Add(new MissedEvent
            {
                Type = MissedEventType.ChatMessage,
                Data = chat
            });
        }
        
        // 3. 系统公告
        var notices = await NoticeManager.GetNoticesAsync(since);
        foreach (var notice in notices)
        {
            events.Add(new MissedEvent
            {
                Type = MissedEventType.SystemNotice,
                Data = notice
            });
        }
        
        return events;
    }
    
    /// <summary>
    /// 发送离线期间错过的消息
    /// </summary>
    private async Task SendMissedEvents(Session session, List<MissedEvent> events)
    {
        // 批量发送（避免单条发送的开销）
        var batch = new S2C_ReconnectSync
        {
            Events = events.Select(e => new MissedEventData
            {
                Type = e.Type,
                Data = JsonSerializer.Serialize(e.Data)
            }).ToList()
        };
        
        await session.SendAsync(batch);
    }
}
```

```csharp
/// <summary>
/// 玩家状态管理器
/// 
/// 注意：此类为设计示例，当前代码库中不存在此类
/// 
/// 职责：
/// 1. 保存玩家状态（实时 + 断线时）
/// 2. 恢复玩家状态
/// 3. 管理状态保留期
/// </summary>
public class PlayerStateManager
{
    // ============================================================
    // 字段
    // ============================================================
    
    /// <summary>
    /// Redis 连接
    /// </summary>
    private readonly IRedis _redis;
    
    /// <summary>
    /// 玩家状态保留期（秒）
    /// </summary>
    private const int StateRetentionSeconds = 300;
    
    /// <summary>
    /// 缓存的在线玩家状态
    /// Key: PlayerId
    /// Value: PlayerState
    /// </summary>
    private readonly ConcurrentDictionary<long, PlayerState> _onlineStates = new();
    
    // ============================================================
    // 状态同步（实时）
    // ============================================================
    
    /// <summary>
    /// 记录玩家状态变化
    /// 建议每 10 秒或关键状态变化时调用
    /// </summary>
    public async Task SyncStateAsync(long playerId, PlayerState state)
    {
        // 更新内存缓存
        _onlineStates[playerId] = state;
        
        // 异步保存到 Redis（降低数据库压力）
        var key = GetStateKey(playerId);
        var json = JsonSerializer.Serialize(state);
        
        // 设置较短过期时间，避免状态永远残留
        await _redis.SetAsync(key, json, 
            TimeSpan.FromSeconds(StateRetentionSeconds));
    }
    
    /// <summary>
    /// 批量同步状态
    /// 用于定期保存所有在线玩家
    /// </summary>
    public async Task BatchSyncAsync()
    {
        var tasks = _onlineStates.Select(kvp => 
            SyncStateAsync(kvp.Key, kvp.Value));
        
        await Task.WhenAll(tasks);
    }
    
    // ============================================================
    // 状态恢复
    // ============================================================
    
    /// <summary>
    /// 恢复玩家状态
    /// </summary>
    /// <param name="playerId">玩家ID</param>
    /// <returns>玩家状态（如果存在）</returns>
    public async Task<PlayerState> RestoreStateAsync(long playerId)
    {
        // 1. 尝试从内存缓存获取（最快的路径）
        if (_onlineStates.TryGetValue(playerId, out var cachedState))
        {
            Logger.Debug($"玩家 {playerId} 状态从内存缓存恢复");
            return cachedState;
        }
        
        // 2. 从 Redis 获取
        var key = GetStateKey(playerId);
        var json = await _redis.GetAsync(key);
        
        if (!string.IsNullOrEmpty(json))
        {
            var state = JsonSerializer.Deserialize<PlayerState>(json);
            
            if (state != null)
            {
                // 恢复到内存缓存
                _onlineStates[playerId] = state;
                
                // 延长保留期
                await ExtendRetention(playerId, 
                    TimeSpan.FromSeconds(StateRetentionSeconds));
                
                Logger.Info($"玩家 {playerId} 状态从 Redis 恢复");
                return state;
            }
        }
        
        // 3. 从数据库加载（兜底）
        var dbState = await LoadFromDatabaseAsync(playerId);
        
        if (dbState != null)
        {
            _onlineStates[playerId] = dbState;
            Logger.Info($"玩家 {playerId} 状态从数据库恢复");
            return dbState;
        }
        
        Logger.Warning($"玩家 {playerId} 状态恢复失败：找不到任何状态");
        return null;
    }
    
    // ============================================================
    // 保留期管理
    // ============================================================
    
    /// <summary>
    /// 延长玩家状态保留期
    /// </summary>
    /// <param name="playerId">玩家ID</param>
    /// <param name="duration">延长时间</param>
    public async Task ExtendRetention(long playerId, TimeSpan duration)
    {
        var key = GetStateKey(playerId);
        await _redis.ExpireAsync(key, duration);
        
        var retentionKey = GetRetentionKey(playerId);
        await _redis.SetAsync(retentionKey, 
            DateTimeOffset.UtcNow.Add(duration).ToUnixTimeSeconds().ToString(),
            duration);
    }
    
    /// <summary>
    /// 清理离线玩家状态
    /// </summary>
    /// <param name="playerId">玩家ID</param>
    public async Task CleanupStateAsync(long playerId)
    {
        // 从内存移除
        _onlineStates.TryRemove(playerId, out _);
        
        // 从 Redis 移除
        var key = GetStateKey(playerId);
        var retentionKey = GetRetentionKey(playerId);
        
        await _redis.DelAsync(key, retentionKey);
        
        // 保存最终状态到数据库
        var state = await RestoreStateAsync(playerId);
        if (state != null)
        {
            await SaveToDatabaseAsync(state);
        }
        
        Logger.Info($"玩家 {playerId} 状态已清理");
    }
    
    // ============================================================
    // 定时清理任务
    // ============================================================
    
    /// <summary>
    /// 清理过期的离线玩家
    /// 建议每分钟执行一次
    /// </summary>
    public async Task CleanupExpiredPlayersAsync()
    {
        var now = DateTimeOffset.UtcNow.ToUnixTimeSeconds();
        
        // 扫描所有保留期 key
        var retentionKeys = await _redis.KeysAsync("player:retention:*");
        
        var expiredCount = 0;
        
        foreach (var retentionKey in retentionKeys)
        {
            var expiryStr = await _redis.GetAsync(retentionKey);
            
            if (long.TryParse(expiryStr, out var expiryTime))
            {
                if (now > expiryTime)
                {
                    // 提取玩家ID
                    var playerIdStr = retentionKey.Replace("player:retention:", "");
                    
                    if (long.TryParse(playerIdStr, out var playerId))
                    {
                        await CleanupStateAsync(playerId);
                        expiredCount++;
                    }
                }
            }
        }
        
        if (expiredCount > 0)
        {
            Logger.Info($"清理了 {expiredCount} 个过期离线玩家状态");
        }
    }
    
    // ============================================================
    // 辅助方法
    // ============================================================
    
    private string GetStateKey(long playerId) => $"player:state:{playerId}";
    private string GetRetentionKey(long playerId) => $"player:retention:{playerId}";

    {

    }

}
```

### 5.1 离线事件处理

```csharp
/// <summary>
/// 离线期间事件管理器
/// 
/// 注意：此类为设计示例，当前代码库中不存在此类
/// 
/// 职责：
/// 1. 记录离线期间的事件
/// 2. 提供离线事件查询
/// 3. 决定哪些事件需要同步给重连玩家
/// </summary>
public class OfflineEventManager
{
    // ============================================================
    // 配置
    // ============================================================
    
    /// <summary>
    /// 离线事件保留期（秒）
    /// </summary>
    private const int EventRetentionSeconds = 3600; // 1小时
    
    /// <summary>
    /// 最大保留事件数（每个玩家）
    /// </summary>
    private const int MaxEventsPerPlayer = 1000;
    
    // ============================================================
    // 字段
    // ============================================================
    
    private readonly IRedis _redis;
    
    // ============================================================
    // 事件记录
    // ============================================================
    
    /// <summary>
    /// 记录事件
    /// </summary>
    public async Task RecordEventAsync(long playerId, OfflineEvent e)
    {
        var key = GetEventKey(playerId);
        
        var eventJson = JsonSerializer.Serialize(new
        {
            e.Type,
            e.Timestamp,
            e.Data
        });
        
        // 使用 Redis List 存储事件
        await _redis.ListRpPushAsync(key, eventJson);
        
        // 设置过期时间
        await _redis.ExpireAsync(key, TimeSpan.FromSeconds(EventRetentionSeconds));
        
        // 限制列表长度
        var listLength = await _redis.ListLengthAsync(key);
        if (listLength > MaxEventsPerPlayer)
        {
            await _redis.ListLTrimAsync(key, 
                listLength - MaxEventsPerPlayer, -1);
        }
    }
    
    /// <summary>
    /// 获取离线事件
    /// </summary>
    public async Task<List<OfflineEvent>> GetEventsAsync(
        long playerId, 
        DateTimeOffset since)
    {
        var key = GetEventKey(playerId);
        var allEvents = await _redis.ListRangeAsync(key);
        
        var events = new List<OfflineEvent>();
        
        foreach (var eventJson in allEvents)
        {
            var evt = JsonSerializer.Deserialize<OfflineEvent>(eventJson);
            
            if (evt != null && evt.Timestamp > since)
            {
                events.Add(evt);
            }
        }
        
        return events;
    }
    
    // ============================================================
    // 事件类型和过滤
    // ============================================================
    
    /// <summary>
    /// 获取需要同步的事件类型
    /// </summary>
    public IEnumerable<OfflineEventType> GetSyncableEventTypes()
    {
        // 这些类型的事件会在重连时同步给玩家
        yield return OfflineEventType.ChatMessage;
        yield return OfflineEventType.SystemNotice;
        yield return OfflineEventType.PlayerEnter;
        yield return OfflineEventType.PlayerLeave;
        yield return OfflineEventType.SceneEvent;
        yield return OfflineEventType.ItemExpired;
        yield return OfflineEventType.BuffExpired;
        yield return OfflineEventType.QuestUpdate;
    }
    
    /// <summary>
    /// 过滤事件（决定是否同步）
    /// </summary>
    public bool ShouldSyncEvent(OfflineEvent evt)
    {
        // 只同步指定的类型
        return GetSyncableEventTypes().Contains(evt.Type);
    }
    
    private string GetEventKey(long playerId) => $"player:events:{playerId}";
}
```
